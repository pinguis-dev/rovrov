1. **POST /api/v1/posts（作成）**:

   ```typescript
   export async function POST(request: Request) {
     const authUser = await getAuthUser(request);
     if (!authUser) {
       return new Response(
         JSON.stringify({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }),
         { status: 401, headers: { 'Content-Type': 'application/json' } },
       );
     }

     // Idempotency-Key処理
     const idempotencyKey = request.headers.get('Idempotency-Key');
     if (!idempotencyKey) {
       return new Response(
         JSON.stringify({
           error: {
             code: 'MISSING_IDEMPOTENCY_KEY',
             message: 'Idempotency-Key header is required',
           },
         }),
         { status: 400, headers: { 'Content-Type': 'application/json' } },
       );
     }

     // 既存処理確認
     const { data: existingResponse } = await supabase
       .from('idempotency_keys')
       .select('response_data, status_code')
       .eq('key', idempotencyKey)
       .eq('user_id', authUser.id)
       .eq('endpoint', 'POST /api/v1/posts')
       .single();

     if (existingResponse) {
       return new Response(existingResponse.response_data, {
         status: existingResponse.status_code,
         headers: { 'Content-Type': 'application/json' },
       });
     }

     const postData = await request.json();
     const {
       caption,
       media_ids = [],
       pin_id,
       tags = [],
       visibility = 'public',
       map_visibility = 'none',
       status = 'draft',
     } = postData;

     try {
       await supabase.rpc('begin_transaction');

       // バリデーション
       if (caption && caption.length > 2000) {
         throw new Error('Caption must be 2000 characters or less');
       }

       if (media_ids.length === 0 && !caption?.trim()) {
         throw new Error('Post must have either media or caption');
       }

       if (media_ids.length > 10) {
         throw new Error('Maximum 10 media files per post');
       }

       // メディアファイル検証（ownership + status)
       if (media_ids.length > 0) {
         const { data: mediaFiles, error: mediaError } = await supabase
           .from('media_files')
           .select('id, status, user_id')
           .in('id', media_ids)
           .eq('user_id', authUser.id);

         if (mediaError || mediaFiles.length !== media_ids.length) {
           throw new Error('Invalid or unauthorized media files');
         }

         const notReadyFiles = mediaFiles.filter((f) => f.status !== 'ready');
         if (notReadyFiles.length > 0) {
           throw new Error('Some media files are not ready for use');
         }
       }

       // Pinの存在確認
       let displayPoint = null;
       if (pin_id) {
         const { data: pin, error: pinError } = await supabase
           .from('pins')
           .select('location')
           .eq('id', pin_id)
           .single();

         if (pinError || !pin) {
           throw new Error('Invalid pin_id');
         }

         // display_point計算
         displayPoint = calculateDisplayPoint(pin.location, map_visibility);
       }

       // タグ正規化
       const normalizedTags = [
         ...new Set(
           tags
             .map((tag: string) => tag.trim().toLowerCase())
             .filter((tag: string) => tag.length > 0 && tag.length <= 50)
             .slice(0, 10), // 最大10タグ
         ),
       ];

       // 投稿作成
       const postId = crypto.randomUUID();
       const { data: post, error: postError } = await supabase
         .from('posts')
         .insert({
           id: postId,
           user_id: authUser.id,
           caption,
           pin_id,
           display_point: displayPoint,
           visibility,
           map_visibility,
           status,
           published_at: status === 'published' ? new Date().toISOString() : null,
           created_at: new Date().toISOString(),
         })
         .select()
         .single();

       if (postError) throw postError;

       // メディアファイル関連付け
       if (media_ids.length > 0) {
         const mediaAssociations = media_ids.map((mediaId: string, index: number) => ({
           post_id: postId,
           media_id: mediaId,
           display_order: index + 1,
         }));

         const { error: mediaAssocError } = await supabase
           .from('post_media')
           .insert(mediaAssociations);

         if (mediaAssocError) throw mediaAssocError;

         // media_filesのpost_id更新
         const { error: mediaUpdateError } = await supabase
           .from('media_files')
           .update({ post_id: postId, updated_at: new Date().toISOString() })
           .in('id', media_ids);

         if (mediaUpdateError) throw mediaUpdateError;
       }

       // タグ関連付け
       if (normalizedTags.length > 0) {
         // タグ作成・取得（upsert）
         const tagInserts = normalizedTags.map((tag) => ({ name: tag }));
         const { data: tags, error: tagsError } = await supabase
           .from('tags')
           .upsert(tagInserts, { onConflict: 'name', ignoreDuplicates: false })
           .select();

         if (tagsError) throw tagsError;

         // 投稿タグ関連付け
         const postTagAssociations = tags.map((tag) => ({
           post_id: postId,
           tag_id: tag.id,
         }));

         const { error: postTagsError } = await supabase
           .from('post_tags')
           .insert(postTagAssociations);

         if (postTagsError) throw postTagsError;
       }

       await supabase.rpc('commit_transaction');

       // レスポンス準備
       const responseData = {
         id: postId,
         user_id: authUser.id,
         caption,
         pin_id,
         display_point: displayPoint,
         visibility,
         map_visibility,
         status,
         tags: normalizedTags,
         media_count: media_ids.length,
         created_at: post.created_at,
         published_at: post.published_at,
       };

       // Idempotency-Key保存
       await supabase.from('idempotency_keys').insert({
         key: idempotencyKey,
         user_id: authUser.id,
         endpoint: 'POST /api/v1/posts',
         response_data: JSON.stringify(responseData),
         status_code: 201,
         expires_at: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), // 24時間
       });

       return new Response(JSON.stringify(responseData), {
         status: 201,
         headers: { 'Content-Type': 'application/json' },
       });
     } catch (error) {
       await supabase.rpc('rollback_transaction');
       console.error('Post creation error:', error);

       return new Response(
         JSON.stringify({
           error: {
             code: 'POST_CREATION_FAILED',
             message: error.message || 'Failed to create post',
           },
         }),
         {
           status: 400,
           headers: { 'Content-Type': 'application/json' },
         },
       );
     }
   }

   // display_point計算関数
   function calculateDisplayPoint(pinLocation: any, mapVisibility: string) {
     switch (mapVisibility) {
       case 'none':
         return null;
       case 'exact':
         return pinLocation;
       case 'approx_100m':
         // 100m四方のグリッドにスナップ
         const coords = pinLocation.coordinates;
         return {
           type: 'Point',
           coordinates: [
             Math.floor(coords[0] * 1000) / 1000, // 約100mグリッド
             Math.floor(coords[1] * 1000) / 1000,
           ],
         };
       default:
         return null;
     }
   }
   ```

2. **PATCH /api/v1/posts/{post_id}（更新）**:

   ```typescript
   export async function PATCH(request: Request, { params }: { params: { post_id: string } }) {
     const { post_id } = params;
     const authUser = await getAuthUser(request);

     if (!authUser) {
       return new Response(
         JSON.stringify({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }),
         { status: 401, headers: { 'Content-Type': 'application/json' } },
       );
     }

     const updateData = await request.json();
     const { caption, pin_id, visibility, map_visibility, status } = updateData;

     try {
       // 投稿の存在・所有権確認
       const { data: existingPost, error: fetchError } = await supabase
         .from('posts')
         .select('*')
         .eq('id', post_id)
         .eq('user_id', authUser.id)
         .single();

       if (fetchError || !existingPost) {
         return new Response(
           JSON.stringify({ error: { code: 'POST_NOT_FOUND', message: 'Post not found' } }),
           { status: 404, headers: { 'Content-Type': 'application/json' } },
         );
       }

       // 削除済み投稿の更新禁止
       if (existingPost.deleted_at) {
         return new Response(
           JSON.stringify({
             error: { code: 'POST_DELETED', message: 'Cannot update deleted post' },
           }),
           { status: 410, headers: { 'Content-Type': 'application/json' } },
         );
       }

       // 状態遷移検証
       if (status && !isValidStatusTransition(existingPost.status, status)) {
         return new Response(
           JSON.stringify({
             error: {
               code: 'INVALID_STATUS_TRANSITION',
               message: `Cannot transition from ${existingPost.status} to ${status}`,
             },
           }),
           { status: 400, headers: { 'Content-Type': 'application/json' } },
         );
       }

       const updates: any = {
         updated_at: new Date().toISOString(),
       };

       if (caption !== undefined) updates.caption = caption;
       if (visibility !== undefined) updates.visibility = visibility;
       if (map_visibility !== undefined) updates.map_visibility = map_visibility;

       // pin_id変更時のdisplay_point再計算
       if (pin_id !== undefined) {
         if (pin_id === null) {
           updates.pin_id = null;
           updates.display_point = null;
         } else {
           const { data: pin, error: pinError } = await supabase
             .from('pins')
             .select('location')
             .eq('id', pin_id)
             .single();

           if (pinError || !pin) {
             return new Response(
               JSON.stringify({ error: { code: 'INVALID_PIN_ID', message: 'Invalid pin_id' } }),
               { status: 400, headers: { 'Content-Type': 'application/json' } },
             );
           }

           updates.pin_id = pin_id;
           updates.display_point = calculateDisplayPoint(
             pin.location,
             map_visibility || existingPost.map_visibility,
           );
         }
       }

       // status更新時のpublished_at設定
       if (status !== undefined) {
         updates.status = status;
         if (
           existingPost.status === 'draft' &&
           (status === 'published' || status === 'temporary')
         ) {
           updates.published_at = new Date().toISOString();
         }
       }

       // map_visibility変更時のdisplay_point再計算
       if (map_visibility !== undefined && existingPost.pin_id) {
         const { data: pin } = await supabase
           .from('pins')
           .select('location')
           .eq('id', existingPost.pin_id)
           .single();

         if (pin) {
           updates.display_point = calculateDisplayPoint(pin.location, map_visibility);
         }
       }

       // 更新実行
       const { data: updatedPost, error: updateError } = await supabase
         .from('posts')
         .update(updates)
         .eq('id', post_id)
         .select()
         .single();

       if (updateError) throw updateError;

       return new Response(JSON.stringify(updatedPost), {
         headers: { 'Content-Type': 'application/json' },
       });
     } catch (error) {
       console.error('Post update error:', error);
       return new Response(
         JSON.stringify({ error: { code: 'UPDATE_FAILED', message: 'Failed to update post' } }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }

   // 状態遷移検証
   function isValidStatusTransition(currentStatus: string, newStatus: string): boolean {
     const validTransitions = {
       draft: ['published', 'temporary', 'archived'],
       published: ['archived'],
       temporary: ['published', 'archived'],
       archived: ['published', 'temporary'],
     };

     return validTransitions[currentStatus]?.includes(newStatus) || false;
   }
   ```

   ```typescript
   export async function DELETE(request: Request, { params }: { params: { post_id: string } }) {
     const { post_id } = params;
     const authUser = await getAuthUser(request);

     if (!authUser) {
       return new Response(
         JSON.stringify({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }),
         { status: 401, headers: { 'Content-Type': 'application/json' } },
       );
     }

     try {
       // 投稿の存在・所有権確認
       const { data: post, error: fetchError } = await supabase
         .from('posts')
         .select('id, user_id, deleted_at')
         .eq('id', post_id)
         .eq('user_id', authUser.id)
         .single();

       if (fetchError || !post) {
         return new Response(
           JSON.stringify({ error: { code: 'POST_NOT_FOUND', message: 'Post not found' } }),
           { status: 404, headers: { 'Content-Type': 'application/json' } },
         );
       }

       if (post.deleted_at) {
         return new Response(
           JSON.stringify({
             error: { code: 'POST_ALREADY_DELETED', message: 'Post already deleted' },
           }),
           { status: 410, headers: { 'Content-Type': 'application/json' } },
         );
       }

       // 論理削除実行
       const { error: deleteError } = await supabase
         .from('posts')
         .update({
           deleted_at: new Date().toISOString(),
           updated_at: new Date().toISOString(),
         })
         .eq('id', post_id);

       if (deleteError) throw deleteError;

       return new Response(null, { status: 204 });
     } catch (error) {
       console.error('Post deletion error:', error);
       return new Response(
         JSON.stringify({ error: { code: 'DELETE_FAILED', message: 'Failed to delete post' } }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }
   ```

   ```typescript
   export async function GET(request: Request, { params }: { params: { post_id: string } }) {
     const { post_id } = params;
     const authUser = await getAuthUser(request); // 匿名可

     try {
       // RLS経由で投稿取得（visibility制御自動適用）
       const { data: post, error: postError } = await supabase
         .from('posts')
         .select(
           `
           id, user_id, caption, pin_id, display_point,
           visibility, map_visibility, status,
           favorite_count, repost_count,
           created_at, published_at, updated_at,
           profiles!posts_user_id_fkey (
             id, username, display_name, avatar_url
           ),
           pins (
             id, name, address, location, google_place_id
           ),
           media_files (
             id, file_name, mime_type, storage_path, external_id,
             visibility, created_at,
             storage_provider
           ),
           post_tags (
             tags (id, name)
           )
         `,
         )
         .eq('id', post_id)
         .is('deleted_at', null)
         .single();

       if (postError || !post) {
         return new Response(
           JSON.stringify({ error: { code: 'POST_NOT_FOUND', message: 'Post not found' } }),
           { status: 404, headers: { 'Content-Type': 'application/json' } },
         );
       }

       // メディアファイルのvisibility制御
       const filteredMedia = post.media_files.filter((media) => {
         if (!media.visibility || media.visibility === 'inherit') {
           return true; // 投稿のvisibilityに従う（RLSで制御済み）
         }

         // メディア独自のvisibility設定がある場合
         if (media.visibility === 'private' && (!authUser || authUser.id !== post.user_id)) {
           return false;
         }

         if (media.visibility === 'friends') {
           // friends関係チェック（簡易実装、実際はRLSまたは別途チェック）
           return authUser && authUser.id !== post.user_id
             ? checkFriendship(authUser.id, post.user_id)
             : true;
         }

         return true; // public
       });

       // レスポンス整形
       const response = {
         ...post,
         media_files: filteredMedia,
         tags: post.post_tags.map((pt) => pt.tags.name),
         author: post.profiles,
         pin: post.pins,
         is_favorited: authUser ? await checkUserFavorited(authUser.id, post_id) : false,
         is_reposted: authUser ? await checkUserReposted(authUser.id, post_id) : false,
       };

       delete response.profiles;
       delete response.pins;
       delete response.post_tags;

       return new Response(JSON.stringify(response), {
         headers: { 'Content-Type': 'application/json' },
       });
     } catch (error) {
       console.error('Post fetch error:', error);
       return new Response(
         JSON.stringify({ error: { code: 'FETCH_FAILED', message: 'Failed to fetch post' } }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }

   async function checkUserFavorited(userId: string, postId: string): Promise<boolean> {
     const { data } = await supabase
       .from('favorites')
       .select('id')
       .eq('user_id', userId)
       .eq('post_id', postId)
       .single();

     return !!data;
   }

   async function checkUserReposted(userId: string, postId: string): Promise<boolean> {
     const { data } = await supabase
       .from('reposts')
       .select('id')
       .eq('user_id', userId)
       .eq('post_id', postId)
       .single();

     return !!data;
   }
   ```

## 状態遷移図

```
draft → published (公開)
draft → temporary (一時公開、24時間後archived)
draft → archived (下書き保存)
published → archived (公開停止)
temporary → published (正式公開)
temporary → archived (期限前停止)
archived → published (再公開)
archived → temporary (一時再公開)
```

## 注意事項

- Idempotency-Key は24時間で自動削除される設定
- display_point計算はPostGIS関数を活用
- メディアファイルのvisibility設定は投稿のvisibilityより制限的にのみ設定可能
- タグは最大10個、各タグは最大50文字の制限
