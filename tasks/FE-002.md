1. **AuthScreen.tsx** (ディーター・ラムス原則準拠):

   ```typescript
   interface AuthScreenProps {
     onAuthSuccess: (user: User) => void;
   }

   const AuthScreen: React.FC<AuthScreenProps> = ({ onAuthSuccess }) => {
     const insets = useSafeAreaInsets();
     const [email, setEmail] = useState('');
     const [isLoading, setIsLoading] = useState(false);
     const [emailSent, setEmailSent] = useState(false);
     const [error, setError] = useState<string | null>(null);

     // アニメーション値
     const fadeAnim = useRef(new Animated.Value(0)).current;
     const scaleAnim = useRef(new Animated.Value(1)).current;

     // メールアドレスバリデーション
     const validateEmail = (email: string): boolean => {
       const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
       return emailRegex.test(email);
     };

     // マジックリンク送信（エラーハンドリング強化）
     const handleSendMagicLink = async () => {
       if (!email.trim()) {
         setError('メールアドレスを入力してください');
         return;
       }

       if (!validateEmail(email)) {
         setError('正しいメールアドレスの形式で入力してください');
         return;
       }

       setError(null);
       setIsLoading(true);

       // ボタンアニメーション
       Animated.sequence([
         Animated.timing(scaleAnim, { toValue: 0.98, duration: 100, useNativeDriver: true }),
         Animated.timing(scaleAnim, { toValue: 1, duration: 100, useNativeDriver: true }),
       ]).start();

       try {
         const { error } = await supabase.auth.signInWithOtp({
           email: email.trim(),
           options: { emailRedirectTo: 'rovrov://auth/callback' },
         });

         if (error) throw error;

         setEmailSent(true);
         Animated.timing(fadeAnim, {
           toValue: 1,
           duration: 300,
           useNativeDriver: true,
         }).start();
       } catch (error: any) {
         // エラーメッセージマッピング
         if (error.message?.includes('rate limit')) {
           setError('試行回数が多すぎます。後でもう一度お試しください');
         } else if (error.message?.includes('network')) {
           setError('インターネット接続を確認してください');
         } else {
           setError('メール送信に失敗しました。もう一度お試しください');
         }
       } finally {
         setIsLoading(false);
       }
     };
   };
   ```

3. **セッション管理**:

   ```typescript
   import * as SecureStore from 'expo-secure-store';

   const AUTH_KEYS = {
     ACCESS_TOKEN: 'access_token',
     REFRESH_TOKEN: 'refresh_token',
     USER_DATA: 'user_data',
   };

   export const authStorage = {
     async setTokens(accessToken: string, refreshToken: string) {
       await SecureStore.setItemAsync(AUTH_KEYS.ACCESS_TOKEN, accessToken);
       await SecureStore.setItemAsync(AUTH_KEYS.REFRESH_TOKEN, refreshToken);
     },

     async getTokens() {
       const accessToken = await SecureStore.getItemAsync(AUTH_KEYS.ACCESS_TOKEN);
       const refreshToken = await SecureStore.getItemAsync(AUTH_KEYS.REFRESH_TOKEN);
       return { accessToken, refreshToken };
     },

     async clearTokens() {
       await SecureStore.deleteItemAsync(AUTH_KEYS.ACCESS_TOKEN);
       await SecureStore.deleteItemAsync(AUTH_KEYS.REFRESH_TOKEN);
       await SecureStore.deleteItemAsync(AUTH_KEYS.USER_DATA);
     },
   };
   ```

### リスクと緩和策

#### 1. セキュリティリスク

**リスク**: トークン漏洩による不正アクセス

- **緩和策**:
  - SecureStoreによる暗号化保存
  - トークン有効期限の短縮（1時間）
  - リフレッシュトークンの安全な管理
  - ディープリンクの署名検証

#### 2. UXリスク

**リスク**: メール遅延による認証フロー中断

- **緩和策**:
  - 明確な待機画面と説明文
  - 再送信オプションの提供（レート制限付き）
  - メールが届かない場合のヘルプリンク

#### 3. 技術的リスク

**リスク**: プラットフォーム間のディープリンク動作差異

- **緩和策**:
  - iOS/Android個別の実装とテスト
  - フォールバックURL（Web経由）の準備
