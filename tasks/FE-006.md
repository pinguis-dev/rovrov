1. **LocationPicker.tsx**:

   ```typescript
   import React, { useState, useEffect } from 'react';
   import { View, Text, Alert, TouchableOpacity, ScrollView } from 'react-native';
   import * as Location from 'expo-location';
   import { Slider } from '@react-native-community/slider';
   import MapView, { Marker } from 'react-native-maps';
   import { PinSelector } from './PinSelector';
   import { Button, Icon, Modal } from '../ui';
   import { locationUtils } from '../../utils/location';
   
   interface LocationPickerProps {
     draft: PostDraft;
     onComplete: (data: { pinId?: string; location?: LocationData }) => void;
     onUpdate: (updates: Partial<PostDraft>) => void;
   }
   
   interface LocationData {
     latitude: number;
     longitude: number;
     address?: string;
   }
   
   interface Pin {
     id: string;
     name: string;
     address: string;
     location: {
       type: 'Point';
       coordinates: [number, number]; // [longitude, latitude]
     };
     distance?: number;
   }
   
   const MAP_VISIBILITY_OPTIONS = [
     {
       value: 'none',
       label: '位置情報を表示しない',
       description: '投稿に位置情報は表示されません'
     },
     {
       value: 'approx_100m',
       label: 'おおよその位置（100m程度）',
       description: 'プライバシーを保護するため、100m程度の範囲で位置を表示'
     },
     {
       value: 'exact',
       label: '正確な位置',
       description: '選択したPinの正確な位置を表示'
     }
   ];
   
   export const LocationPicker: React.FC<LocationPickerProps> = ({ 
     draft, 
     onComplete, 
     onUpdate 
   }) => {
     const [currentLocation, setCurrentLocation] = useState<LocationData | null>(null);
     const [selectedPin, setSelectedPin] = useState<Pin | null>(null);
     const [mapVisibility, setMapVisibility] = useState<string>(draft.mapVisibility || 'none');
     const [nearbyPins, setNearbyPins] = useState<Pin[]>([]);
     const [isLoadingLocation, setIsLoadingLocation] = useState(false);
     const [showPinSelector, setShowPinSelector] = useState(false);
     const [showPrivacyInfo, setShowPrivacyInfo] = useState(false);
     
     useEffect(() => {
       // EXIF位置情報があれば初期化
       if (draft.location) {
         setCurrentLocation(draft.location);
         searchNearbyPins(draft.location);
       }
       
       // 選択済みPinがあれば設定
       if (draft.pinId) {
         // Pin詳細取得は別途実装
       }
     }, [draft]);
     
     const requestLocationPermission = async () => {
       const { status } = await Location.requestForegroundPermissionsAsync();
       if (status !== 'granted') {
         Alert.alert('位置情報の権限が必要です', '現在地を取得するには位置情報へのアクセス権限を許可してください');
         return false;
       }
       return true;
     };
     
     const getCurrentLocation = async () => {
       const hasPermission = await requestLocationPermission();
       if (!hasPermission) return;
       
       setIsLoadingLocation(true);
       try {
         const location = await Location.getCurrentPositionAsync({
           accuracy: Location.Accuracy.High,
           timeoutMs: 10000,
         });
         
         const locationData: LocationData = {
           latitude: location.coords.latitude,
           longitude: location.coords.longitude
         };
         
         // 住所を逆ジオコーディング
         try {
           const addresses = await Location.reverseGeocodeAsync({
             latitude: locationData.latitude,
             longitude: locationData.longitude
           });
           
           if (addresses.length > 0) {
             const addr = addresses[0];
             locationData.address = `${addr.region || ''} ${addr.city || ''} ${addr.street || ''}`.trim();
           }
         } catch (error) {
           console.warn('Reverse geocoding failed:', error);
         }
         
         setCurrentLocation(locationData);
         onUpdate({ location: locationData });
         
         // 近傍Pin検索
         await searchNearbyPins(locationData);
         
       } catch (error) {
         console.error('Location acquisition error:', error);
         Alert.alert('エラー', '現在地を取得できませんでした');
       } finally {
         setIsLoadingLocation(false);
       }
     };
     
     const searchNearbyPins = async (location: LocationData) => {
       try {
         // 1km以内のPin検索（仮のAPI実装）
         const response = await fetch(
           `/api/v1/pins/nearby?lat=${location.latitude}&lng=${location.longitude}&radius=1000`
         );
         const data = await response.json();
         
         // 距離計算
         const pinsWithDistance = data.pins.map((pin: Pin) => ({
           ...pin,
           distance: locationUtils.calculateDistance(
             location.latitude,
             location.longitude,
             pin.location.coordinates[1], // latitude
             pin.location.coordinates[0]  // longitude
           )
         }));
         
         pinsWithDistance.sort((a, b) => (a.distance || 0) - (b.distance || 0));
         setNearbyPins(pinsWithDistance);
         
       } catch (error) {
         console.error('Nearby pins search error:', error);
       }
     };
     
     const handlePinSelect = (pin: Pin) => {
       setSelectedPin(pin);
       onUpdate({ pinId: pin.id });
       setShowPinSelector(false);
     };
     
     const handleMapVisibilityChange = (value: string) => {
       setMapVisibility(value);
       onUpdate({ mapVisibility: value });
     };
     
     const getDisplayPoint = () => {
       if (!selectedPin || mapVisibility === 'none') return null;
       
       const pinCoords = selectedPin.location.coordinates;
       
       switch (mapVisibility) {
         case 'exact':
           return {
             latitude: pinCoords[1],
             longitude: pinCoords[0]
           };
         case 'approx_100m':
           return locationUtils.snapToGrid(pinCoords[1], pinCoords[0], 0.001); // 約100m
         default:
           return null;
       }
     };
     
     const handleNext = () => {
       if (mapVisibility !== 'none' && !selectedPin) {
         Alert.alert('確認', '位置情報を表示するにはPinを選択してください');
         return;
       }
       
       onComplete({
         pinId: selectedPin?.id,
         location: currentLocation
       });
     };
     
     const displayPoint = getDisplayPoint();
     
     return (
       <ScrollView className="flex-1 p-4">
         <Text className="text-xl font-bold mb-4">位置情報設定</Text>
         
         {/* 現在地取得 */}
         <View className="mb-6">
           <Text className="text-lg font-medium mb-2">現在地</Text>
           <Button
             title={currentLocation ? '現在地を更新' : '現在地を取得'}
             onPress={getCurrentLocation}
             loading={isLoadingLocation}
             icon="map-pin"
           />
           {currentLocation && (
             <Text className="text-sm text-gray-600 mt-2">
               {currentLocation.address || `${currentLocation.latitude.toFixed(4)}, ${currentLocation.longitude.toFixed(4)}`}
             </Text>
           )}
         </View>
         
         {/* Pin選択 */}
         <View className="mb-6">
           <View className="flex-row items-center justify-between mb-2">
             <Text className="text-lg font-medium">Pin選択</Text>
             <TouchableOpacity onPress={() => setShowPinSelector(true)}>
               <Text className="text-blue-500">詳細選択</Text>
             </TouchableOpacity>
           </View>
           
           {selectedPin ? (
             <View className="bg-gray-100 p-3 rounded-lg">
               <Text className="font-medium">{selectedPin.name}</Text>
               <Text className="text-sm text-gray-600">{selectedPin.address}</Text>
               {selectedPin.distance && (
                 <Text className="text-xs text-gray-500 mt-1">
                   約{Math.round(selectedPin.distance)}m
                 </Text>
               )}
             </View>
           ) : (
             <TouchableOpacity
               className="bg-gray-100 p-3 rounded-lg items-center"
               onPress={() => setShowPinSelector(true)}
             >
               <Text className="text-gray-600">Pinを選択してください</Text>
             </TouchableOpacity>
           )}
         </View>
         
         {/* 地図上の位置表示設定 */}
         <View className="mb-6">
           <View className="flex-row items-center justify-between mb-3">
             <Text className="text-lg font-medium">地図上の位置表示</Text>
             <TouchableOpacity onPress={() => setShowPrivacyInfo(true)}>
               <Icon name="info" size={20} color="#666" />
             </TouchableOpacity>
           </View>
           
           {MAP_VISIBILITY_OPTIONS.map((option) => (
             <TouchableOpacity
               key={option.value}
               className={`p-3 rounded-lg mb-2 border ${
                 mapVisibility === option.value 
                   ? 'border-blue-500 bg-blue-50' 
                   : 'border-gray-200 bg-white'
               }`}
               onPress={() => handleMapVisibilityChange(option.value)}
             >
               <Text className="font-medium">{option.label}</Text>
               <Text className="text-sm text-gray-600 mt-1">{option.description}</Text>
             </TouchableOpacity>
           ))}
         </View>
         
         {/* プレビューマップ */}
         {displayPoint && (
           <View className="mb-6">
             <Text className="text-lg font-medium mb-2">プレビュー</Text>

                 zoomEnabled={false}
               >
                 <Marker
                   coordinate={displayPoint}

                 />
               </MapView>

               {mapVisibility === 'approx_100m'
   ```

   ```typescript
   import React, { useState, useEffect } from 'react';
   import { View, Text, FlatList, TouchableOpacity, TextInput, Alert } from 'react-native';
   import * as Location from 'expo-location';
   import { Button, Icon, LoadingSpinner } from '../ui';
   import { pinApi } from '../../services/pinApi';
   import { googlePlacesApi } from '../../services/googlePlacesApi';
   import { locationUtils } from '../../utils/location';
   import { generateIdempotencyKey } from '../../utils/api';
   
   interface PinSelectorProps {
     currentLocation: LocationData | null;
     nearbyPins: Pin[];
     selectedPin: Pin | null;
     onSelect: (pin: Pin) => void;
     onClose: () => void;
   }
   
   interface GooglePlace {
     place_id: string;
     name: string;
     formatted_address: string;
     geometry: {
       location: {
         lat: number;
         lng: number;
       };
     };
     types: string[];
   }
   
   export const PinSelector: React.FC<PinSelectorProps> = ({
     currentLocation,
     nearbyPins,
     selectedPin,
     onSelect,
     onClose
   }) => {
     const [searchQuery, setSearchQuery] = useState('');
     const [searchResults, setSearchResults] = useState<GooglePlace[]>([]);
     const [isSearching, setIsSearching] = useState(false);
     const [showCreateForm, setShowCreateForm] = useState(false);
     const [newPinName, setNewPinName] = useState('');
     const [newPinAddress, setNewPinAddress] = useState('');
     const [isCreating, setIsCreating] = useState(false);
     
     const searchPlaces = async (query: string) => {
       if (!query.trim() || !currentLocation) return;
       
       setIsSearching(true);
       try {
         const places = await googlePlacesApi.searchPlaces({
           query,
           location: currentLocation,
           radius: 5000 // 5km
         });
         
         setSearchResults(places);
       } catch (error) {
         console.error('Places search error:', error);
         Alert.alert('エラー', '場所の検索に失敗しました');
       } finally {
         setIsSearching(false);
       }
     };
     
     const createPinFromPlace = async (place: GooglePlace) => {
       setIsCreating(true);
       try {
         const pinData = {
           name: place.name,
           address: place.formatted_address,
           location: {
             type: 'Point',
             coordinates: [place.geometry.location.lng, place.geometry.location.lat]
           },
           google_place_id: place.place_id,
           is_google_place: true
         };
         
         const createdPin = await pinApi.createPin(pinData, generateIdempotencyKey());
         onSelect(createdPin);
         
       } catch (error: any) {
         console.error('Pin creation error:', error);
         Alert.alert('エラー', error.response?.data?.error?.message || 'Pinの作成に失敗しました');
       } finally {
         setIsCreating(false);
       }
     };
     
     const createCustomPin = async () => {
       if (!newPinName.trim() || !currentLocation) return;
       
       setIsCreating(true);
       try {
         const pinData = {
           name: newPinName.trim(),
           address: newPinAddress.trim() || undefined,
           location: {
             type: 'Point',
             coordinates: [currentLocation.longitude, currentLocation.latitude]
           },
           is_google_place: false
         };
         
         const createdPin = await pinApi.createPin(pinData, generateIdempotencyKey());
         onSelect(createdPin);
         setShowCreateForm(false);
         setNewPinName('');
         setNewPinAddress('');
         
       } catch (error: any) {
         console.error('Custom pin creation error:', error);
         Alert.alert('エラー', error.response?.data?.error?.message || 'Pinの作成に失敗しました');
       } finally {
         setIsCreating(false);
       }
     };
     
     const renderNearbyPin = ({ item }: { item: Pin }) => (
       <TouchableOpacity
         className={`p-3 border rounded-lg mb-2 ${
           selectedPin?.id === item.id 
             ? 'border-blue-500 bg-blue-50' 
             : 'border-gray-200 bg-white'
         }`}
         onPress={() => onSelect(item)}
       >
         <Text className="font-medium">{item.name}</Text>
         <Text className="text-sm text-gray-600 mt-1">{item.address}</Text>
         {item.distance && (
           <Text className="text-xs text-gray-500 mt-1">約{Math.round(item.distance)}m</Text>
         )}
       </TouchableOpacity>
     );
     
     const renderSearchResult = ({ item }: { item: GooglePlace }) => (
       <TouchableOpacity
         className="p-3 border border-gray-200 rounded-lg mb-2 bg-white"
         onPress={() => createPinFromPlace(item)}
         disabled={isCreating}
       >
         <View className="flex-row items-start">
           <View className="flex-1">
             <Text className="font-medium">{item.name}</Text>
             <Text className="text-sm text-gray-600 mt-1">{item.formatted_address}</Text>
             <Text className="text-xs text-gray-500 mt-1">
               {item.types.slice(0, 3).join(', ')}
             </Text>
           </View>
           <Icon name="plus" size={20} color="#666" />
         </View>
       </TouchableOpacity>
     );
     
     return (
       <View className="flex-1">
         {/* 検索バー */}
         <View className="mb-4">
           <TextInput
             className="border border-gray-300 rounded-lg p-3"
             placeholder="場所を検索..."
             value={searchQuery}
             onChangeText={setSearchQuery}
             onSubmitEditing={() => searchPlaces(searchQuery)}
             returnKeyType="search"
           />
           <Button
             title="検索"
             onPress={() => searchPlaces(searchQuery)}
             loading={isSearching}
             className="mt-2"
           />
         </View>
         
         {/* タブ切り替え */}
         <View className="flex-row mb-4">
           <TouchableOpacity
             className="flex-1 p-2 border-b-2 border-blue-500"
             onPress={() => setShowCreateForm(false)}
           >
             <Text className="text-center font-medium">近くのPin</Text>
           </TouchableOpacity>
           <TouchableOpacity
             className="flex-1 p-2 border-b-2 border-gray-200"
             onPress={() => setShowCreateForm(true)}
           >
             <Text className="text-center font-medium text-gray-600">新規作成</Text>
           </TouchableOpacity>
         </View>
         
         {/* 内容エリア */}
         <View className="flex-1">
           {!showCreateForm ? (
             <>
               {/* 検索結果 */}
               {searchResults.length > 0 && (
                 <View className="mb-4">
                   <Text className="font-medium mb-2">検索結果</Text>
                   <FlatList
                     data={searchResults}
                     renderItem={renderSearchResult}
                     keyExtractor={(item) => item.place_id}
                     maxHeight={200}
                   />
                 </View>
               )}
               
               {/* 近傍Pin */}
               <Text className="font-medium mb-2">
                 近くのPin ({nearbyPins.length}件)
               </Text>
               <FlatList
                 data={nearbyPins}
                 renderItem={renderNearbyPin}
                 keyExtractor={(item) => item.id}
                 showsVerticalScrollIndicator={false}
               />
             </>
           ) : (
             /* 新規作成フォーム */
             <View>
               <Text className="font-medium mb-4">新しいPinを作成</Text>
               <TextInput
                 className="border border-gray-300 rounded-lg p-3 mb-3"
                 placeholder="Pin名（必須）"
                 value={newPinName}
                 onChangeText={setNewPinName}
                 maxLength={50}
               />
               <TextInput
                 className="border border-gray-300 rounded-lg p-3 mb-4"
                 placeholder="住所（任意）"
                 value={newPinAddress}
                 onChangeText={setNewPinAddress}
                 maxLength={200}
                 multiline
                 numberOfLines={3}
               />
               
               <Button
                 title="Pinを作成"
                 onPress={createCustomPin}
                 disabled={!newPinName.trim() || !currentLocation}
                 loading={isCreating}
               />
               
               {!currentLocation && (
                 <Text className="text-sm text-red-500 mt-2">
                   現在地を取得してからPinを作成してください
                 </Text>
               )}
             </View>
           )}
         </View>
         
         {/* フッター */}
         <View className="pt-4 border-t border-gray-200">
           <Button
             title="閉じる"
             variant="outline"
             onPress={onClose}
           />
         </View>
       </View>
     );
   };
   ```

   ```typescript
   // utils/location.ts
   export const locationUtils = {
     // 2点間の距離計算（メートル）
     calculateDistance(lat1: number, lng1: number, lat2: number, lng2: number): number {
       const R = 6371000; // 地球の半径（メートル）
       const dLat = this.toRadians(lat2 - lat1);
       const dLng = this.toRadians(lng2 - lng1);
       const a = 
         Math.sin(dLat / 2) * Math.sin(dLat / 2) +
         Math.cos(this.toRadians(lat1)) * Math.cos(this.toRadians(lat2)) *
         Math.sin(dLng / 2) * Math.sin(dLng / 2);
       const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
       return R * c;
     },
     
     // 座標をグリッドにスナップ（プライバシー保護用）
     snapToGrid(lat: number, lng: number, gridSize: number = 0.001) {
       return {
         latitude: Math.floor(lat / gridSize) * gridSize + (gridSize / 2),
         longitude: Math.floor(lng / gridSize) * gridSize + (gridSize / 2)
       };
     },
     
     // 度をラジアンに変換
     toRadians(degrees: number): number {
       return degrees * (Math.PI / 180);
     },
     
     // 座標の精度を制限
     limitPrecision(coord: number, precision: number = 4): number {
       return Math.round(coord * Math.pow(10, precision)) / Math.pow(10, precision);
     }
   };
   ```
