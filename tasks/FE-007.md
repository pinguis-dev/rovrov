   ```typescript
   import React, { useState, useEffect } from 'react';
   import { View, Text, FlatList, RefreshControl, Alert } from 'react-native';
   import { useInfiniteQuery } from '@tanstack/react-query';
   import { TabView, SceneMap, TabBar } from 'react-native-tab-view';
   import { useAuthStore } from '../store/authStore';
   import { timelineApi } from '../services/timelineApi';
   import { PostCard } from '../components/timeline/PostCard';
   import { LoadingSpinner, EmptyState } from '../components/ui';
   
   interface TimelineTab {
     key: string;
     title: string;
     requiresAuth: boolean;
   }
   
   const TIMELINE_TABS: TimelineTab[] = [
     { key: 'featured', title: 'Featured', requiresAuth: false },
     { key: 'follow', title: 'Follow', requiresAuth: true },
     { key: 'friend', title: 'Friends', requiresAuth: true }
   ];
   
   export const TimelineScreen: React.FC = () => {
     const { user, isAuthenticated } = useAuthStore();
     const [tabIndex, setTabIndex] = useState(0);
     
     // 認証が必要なタブの場合、Featuredに戻す
     useEffect(() => {
       if (!isAuthenticated && TIMELINE_TABS[tabIndex].requiresAuth) {
         setTabIndex(0);
       }
     }, [isAuthenticated, tabIndex]);
     
     const renderTabBar = (props: any) => (
       <TabBar
         {...props}
         indicatorStyle={{ backgroundColor: '#3B82F6' }}
         style={{ backgroundColor: 'white', elevation: 0, shadowOpacity: 0 }}
         labelStyle={{ color: '#374151', fontWeight: '600' }}
         activeColor="#3B82F6"
         inactiveColor="#9CA3AF"
       />
     );
     
     const FeaturedTimeline = () => <TimelineTab type="featured" />;
     const FollowTimeline = () => isAuthenticated ? <TimelineTab type="follow" /> : <AuthRequiredState />;
     const FriendTimeline = () => isAuthenticated ? <TimelineTab type="friend" /> : <AuthRequiredState />;
     
     const renderScene = SceneMap({
       featured: FeaturedTimeline,
       follow: FollowTimeline,
       friend: FriendTimeline
     });
     
     const routes = TIMELINE_TABS.map(tab => ({
       key: tab.key,
       title: tab.title
     }));
     
     return (
       <View className="flex-1 bg-white">
         <TabView
           navigationState={{ index: tabIndex, routes }}
           renderScene={renderScene}
           renderTabBar={renderTabBar}
           onIndexChange={setTabIndex}
           lazy
         />
       </View>
     );
   };
   
   interface TimelineTabProps {
     type: 'featured' | 'follow' | 'friend';
   }
   
   const TimelineTab: React.FC<TimelineTabProps> = ({ type }) => {
     const {
       data,
       fetchNextPage,
       hasNextPage,
       isFetchingNextPage,
       refetch,
       isRefetching,
       isLoading,
       error
     } = useInfiniteQuery({
       queryKey: ['timeline', type],
       queryFn: ({ pageParam }) => timelineApi.getTimeline(type, { cursor: pageParam }),
       getNextPageParam: (lastPage) => lastPage.next_cursor,
       staleTime: 5 * 60 * 1000, // 5分
       cacheTime: 30 * 60 * 1000, // 30分
     });
     
     const posts = data?.pages.flatMap(page => page.posts) || [];
     
     const handleLoadMore = () => {
       if (hasNextPage && !isFetchingNextPage) {
         fetchNextPage();
       }
     };
     
     const handleRefresh = () => {
       refetch();
     };
     
     if (isLoading) {
       return (
         <View className="flex-1 items-center justify-center">
           <LoadingSpinner size="large" />
         </View>
       );
     }
     
     if (error) {
       return (
         <View className="flex-1 items-center justify-center p-4">
           <Text className="text-red-600 text-center mb-4">
             タイムラインの読み込みに失敗しました
           </Text>
           <TouchableOpacity
             className="bg-blue-500 px-4 py-2 rounded-lg"
             onPress={() => refetch()}
           >
             <Text className="text-white font-medium">再試行</Text>
           </TouchableOpacity>
         </View>
       );
     }
     
     if (posts.length === 0) {
       return (
         <EmptyState
           title="投稿がありません"
           description={getEmptyStateDescription(type)}
           action={type === 'featured' ? undefined : {
             title: '投稿を作成',
             onPress: () => navigation.navigate('PostCreator')
           }}
         />
       );
     }
     
     const renderPost = ({ item, index }: { item: Post; index: number }) => {
       // 日付グルーピング
       const showDateHeader = index === 0 || 
         !isSameDay(new Date(item.published_at), new Date(posts[index - 1].published_at));
       
       return (
         <View>
           {showDateHeader && (
             <View className="bg-gray-50 px-4 py-2 border-b border-gray-100">
               <Text className="text-sm font-medium text-gray-600">
                 {formatDateHeader(item.published_at)}
               </Text>
             </View>
           )}
           <PostCard post={item} />
         </View>
       );
     };
     
     const renderFooter = () => {
       if (!isFetchingNextPage) return null;
       
       return (
         <View className="py-4 items-center">
           <LoadingSpinner />
         </View>
       );
     };
     
     return (
       <FlatList
         data={posts}
         renderItem={renderPost}
         keyExtractor={(item) => item.id}
         refreshControl={
           <RefreshControl
             refreshing={isRefetching}
             onRefresh={handleRefresh}
             tintColor="#3B82F6"
           />
         }
         onEndReached={handleLoadMore}
         onEndReachedThreshold={0.8}
         ListFooterComponent={renderFooter}
         removeClippedSubviews
         maxToRenderPerBatch={10}
         windowSize={10}
         getItemLayout={(data, index) => ({
           length: ESTIMATED_POST_HEIGHT,
           offset: ESTIMATED_POST_HEIGHT * index,
           index
         })}
       />
     );
   };
   ```

   ```typescript
   import React, { useState, useRef } from 'react';
   import { View, Text, TouchableOpacity, Dimensions, Image } from 'react-native';
   import { Video } from 'expo-av';
   import Carousel from 'react-native-snap-carousel';
   import { useNavigation } from '@react-navigation/native';
   import { useMutation, useQueryClient } from '@tanstack/react-query';
   import { socialApi } from '../../services/socialApi';
   import { Icon, Avatar } from '../ui';
   import { formatRelativeTime, formatNumber } from '../../utils/format';
   
   const { width: screenWidth } = Dimensions.get('window');
   const MEDIA_WIDTH = screenWidth - 32; // padding考慮
   const MEDIA_HEIGHT = MEDIA_WIDTH * 0.75; // 4:3 aspect ratio
   
   interface PostCardProps {
     post: Post;
   }
   
   export const PostCard: React.FC<PostCardProps> = ({ post }) => {
     const navigation = useNavigation();
     const queryClient = useQueryClient();
     const [currentMediaIndex, setCurrentMediaIndex] = useState(0);
     const videoRef = useRef<Video>(null);
     
     // オプティミスティック更新用
     const favoriteMutation = useMutation({
       mutationFn: (action: 'add' | 'remove') => 
         action === 'add' 
           ? socialApi.favoritePost(post.id)
           : socialApi.unfavoritePost(post.id),
       onMutate: async (action) => {
         // オプティミスティック更新
         await queryClient.cancelQueries({ queryKey: ['timeline'] });
         
         const previousData = queryClient.getQueriesData({ queryKey: ['timeline'] });
         
         queryClient.setQueriesData(
           { queryKey: ['timeline'] },
           (old: any) => {
             if (!old) return old;
             
             return {
               ...old,
               pages: old.pages.map((page: any) => ({
                 ...page,
                 posts: page.posts.map((p: Post) => 
                   p.id === post.id
                     ? {
                         ...p,
                         is_favorited: action === 'add',
                         favorite_count: p.favorite_count + (action === 'add' ? 1 : -1)
                       }
                     : p
                 )
               }))
             };
           }
         );
         
         return { previousData };
       },
       onError: (error, variables, context) => {
         // エラー時にロールバック
         if (context?.previousData) {
           context.previousData.forEach(([queryKey, data]) => {
             queryClient.setQueryData(queryKey, data);
           });
         }
         Alert.alert('エラー', 'いいねの処理に失敗しました');
       }
     });
     
     const repostMutation = useMutation({
       mutationFn: (action: 'add' | 'remove') => 
         action === 'add' 
           ? socialApi.repostPost(post.id)
           : socialApi.unrepostPost(post.id),
       // Similar optimistic update logic...
     });
     
     const handleUserPress = () => {
       navigation.navigate('Profile', { userId: post.user_id });
     };
     
     const handlePostPress = () => {
       navigation.navigate('PostDetail', { postId: post.id });
     };
     
     const handleFavorite = () => {
       favoriteMutation.mutate(post.is_favorited ? 'remove' : 'add');
     };
     
     const handleRepost = () => {
       Alert.alert(
         'リポスト',
         'この投稿をリポストしますか？',
         [
           { text: 'キャンセル', style: 'cancel' },
           { 
             text: 'リポスト', 
             onPress: () => repostMutation.mutate(post.is_reposted ? 'remove' : 'add')
           }
         ]
       );
     };
     
     const renderMediaItem = ({ item, index }: { item: MediaFile; index: number }) => {
       const isVideo = item.mime_type.startsWith('video/');
       
       return (
         <View style={{ width: MEDIA_WIDTH, height: MEDIA_HEIGHT }}>
           {isVideo ? (
             <Video
               ref={index === currentMediaIndex ? videoRef : null}
               source={{ uri: item.url }}
               style={{ width: '100%', height: '100%', borderRadius: 12 }}
               shouldPlay={false}
               isLooping
               isMuted
               resizeMode="cover"
               useNativeControls
             />
           ) : (
             <Image
               source={{ uri: item.url }}
               style={{ width: '100%', height: '100%', borderRadius: 12 }}
               resizeMode="cover"
             />
           )}
         </View>
       );
     };
     
     return (
       <TouchableOpacity 
         className="bg-white border-b border-gray-100" 
         onPress={handlePostPress}
         activeOpacity={0.95}
       >
         <View className="p-4">
           {/* ユーザー情報 */}
           <TouchableOpacity 
             className="flex-row items-center mb-3"
             onPress={handleUserPress}
           >
             <Avatar 
               source={{ uri: post.author.avatar_url }}
               size="medium"
               fallback={post.author.display_name?.charAt(0) || post.author.username.charAt(0)}
             />
             <View className="ml-3 flex-1">
               <Text className="font-semibold text-gray-900">
                 {post.author.display_name || post.author.username}
               </Text>
               <Text className="text-sm text-gray-500">
                 @{post.author.username} · {formatRelativeTime(post.published_at)}
               </Text>
             </View>
           </TouchableOpacity>
           
           {/* キャプション */}
           {post.caption && (
             <Text className="text-gray-900 mb-3 text-base leading-relaxed">
               {post.caption}
             </Text>
           )}
           
           {/* メディア */}
           {post.media_files.length > 0 && (
             <View className="mb-3">
               {post.media_files.length === 1 ? (
                 renderMediaItem({ item: post.media_files[0], index: 0 })
               ) : (
                 <View>
                   <Carousel
                     data={post.media_files}
                     renderItem={renderMediaItem}
                     sliderWidth={MEDIA_WIDTH}
                     itemWidth={MEDIA_WIDTH}
                     onSnapToItem={setCurrentMediaIndex}
                     enableMomentum={false}
                     decelerationRate={0.9}
                   />
                   <View className="flex-row justify-center mt-2">
                     {post.media_files.map((_, index) => (
                       <View
                         key={index}
                         className={`w-2 h-2 rounded-full mx-1 ${
                           index === currentMediaIndex ? 'bg-blue-500' : 'bg-gray-300'
                         }`}
                       />
                     ))}
                   </View>
                 </View>
               )}
             </View>
           )}
           
           {/* 位置情報 */}
           {post.pin && (
             <TouchableOpacity 
               className="flex-row items-center mb-3"
               onPress={() => navigation.navigate('Map', { pinId: post.pin.id })}
             >
               <Icon name="map-pin" size={16} color="#6B7280" />
               <Text className="text-sm text-gray-600 ml-1">
                 {post.pin.name}
               </Text>
             </TouchableOpacity>
           )}
           
           {/* アクションボタン */}
           <View className="flex-row items-center justify-between">
             <TouchableOpacity 
               className="flex-row items-center"
               onPress={handleFavorite}
               disabled={favoriteMutation.isLoading}
             >
               <Icon 
                 name={post.is_favorited ? 'heart' : 'heart-outline'}
                 size={20}
                 color={post.is_favorited ? '#EF4444' : '#6B7280'}
               />
               {post.favorite_count > 0 && (
                 <Text className="text-sm text-gray-600 ml-1">
                   {formatNumber(post.favorite_count)}
                 </Text>
               )}
             </TouchableOpacity>
             
             <TouchableOpacity 
               className="flex-row items-center"
               onPress={handleRepost}
               disabled={repostMutation.isLoading}
             >
               <Icon 
                 name={post.is_reposted ? 'repeat' : 'repeat-outline'}
                 size={20}
                 color={post.is_reposted ? '#10B981' : '#6B7280'}
               />
               {post.repost_count > 0 && (
                 <Text className="text-sm text-gray-600 ml-1">
                   {formatNumber(post.repost_count)}
                 </Text>
               )}
             </TouchableOpacity>
             
             <TouchableOpacity className="flex-row items-center">
               <Icon name="bookmark-outline" size={20} color="#6B7280" />
             </TouchableOpacity>
             
             <TouchableOpacity className="flex-row items-center">
               <Icon name="share-outline" size={20} color="#6B7280" />
             </TouchableOpacity>
           </View>
         </View>
       </TouchableOpacity>
     );
   };
   ```

   ```typescript
     async getTimeline(
       type: 'featured' | 'follow' | 'friend', 
       params?: { cursor?: string; limit?: number }
     ) {
   ```

```typescript
// utils/format.ts
export const formatRelativeTime = (dateString: string): string => {
  const date = new Date(dateString);
  const now = new Date();
  const diffInSeconds = Math.floor((now.getTime() - date.getTime()) / 1000);
  
  if (diffInSeconds < 60) {
    return '今';
  } else if (diffInSeconds < 3600) {
    return `${Math.floor(diffInSeconds / 60)}分前`;
  } else if (diffInSeconds < 86400) {
    return `${Math.floor(diffInSeconds / 3600)}時間前`;
  } else if (diffInSeconds < 604800) {
    return `${Math.floor(diffInSeconds / 86400)}日前`;
  } else {
    return date.toLocaleDateString('ja-JP', {
      month: 'short',
      day: 'numeric'
    });
  }
};

export const formatNumber = (num: number): string => {
  if (num < 1000) {
    return num.toString();
  } else if (num < 1000000) {
    return `${(num / 1000).toFixed(1)}K`;
  } else {
    return `${(num / 1000000).toFixed(1)}M`;
  }
};

export const formatDateHeader = (dateString: string): string => {
  const date = new Date(dateString);
  const now = new Date();
  
  if (isSameDay(date, now)) {
    return '今日';
  } else if (isSameDay(date, new Date(now.getTime() - 86400000))) {
    return '昨日';
  } else {
    return date.toLocaleDateString('ja-JP', {
      month: 'long',
      day: 'numeric',
      weekday: 'short'
    });
  }
};

export const isSameDay = (date1: Date, date2: Date): boolean => {
  return date1.getFullYear() === date2.getFullYear() &&
         date1.getMonth() === date2.getMonth() &&
         date1.getDate() === date2.getDate();
};
```
