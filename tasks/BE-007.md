1. **POST /api/v1/media/upload-url（画像用）**:

   ```typescript
   import { createClient } from '@supabase/supabase-js';
   import crypto from 'crypto';

   export async function POST(request: Request) {
     const authUser = await getAuthUser(request);
     if (!authUser) {
       return new Response(
         JSON.stringify({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }),
         { status: 401, headers: { 'Content-Type': 'application/json' } },
       );
     }

     const { fileName, fileSize, mimeType } = await request.json();

     // ファイルサイズ検証
     const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
     if (fileSize > MAX_FILE_SIZE) {
       return new Response(
         JSON.stringify({
           error: { code: 'FILE_TOO_LARGE', message: 'File size must be less than 10MB' },
         }),
         { status: 413, headers: { 'Content-Type': 'application/json' } },
       );
     }

     // MIME type検証
     const allowedMimeTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
     if (!allowedMimeTypes.includes(mimeType)) {
       return new Response(
         JSON.stringify({
           error: {
             code: 'INVALID_FILE_TYPE',
             message: 'Invalid file type. Only JPEG, PNG, GIF, and WebP are allowed',
           },
         }),
         { status: 400, headers: { 'Content-Type': 'application/json' } },
       );
     }

     // 拡張子偽装対策
     const validExtensions = {
       'image/jpeg': ['.jpg', '.jpeg'],
       'image/png': ['.png'],
       'image/gif': ['.gif'],
       'image/webp': ['.webp'],
     };

     const fileExtension = fileName.toLowerCase().substring(fileName.lastIndexOf('.'));
     if (!validExtensions[mimeType].includes(fileExtension)) {
       return new Response(
         JSON.stringify({
           error: {
             code: 'EXTENSION_MISMATCH',
             message: 'File extension does not match MIME type',
           },
         }),
         { status: 400, headers: { 'Content-Type': 'application/json' } },
       );
     }

     try {
       // ユニークなファイルパス生成
       const fileId = crypto.randomUUID();
       const sanitizedFileName = fileName.replace(/[^a-zA-Z0-9.-]/g, '');
       const filePath = `posts/${authUser.id}/${fileId}/${sanitizedFileName}`;

       // media_filesレコード作成
       const { data: mediaFile, error: dbError } = await supabase
         .from('media_files')
         .insert({
           id: fileId,
           user_id: authUser.id,
           file_name: sanitizedFileName,
           mime_type: mimeType,
           file_size: fileSize,
           storage_path: filePath,
           status: 'uploading',
           created_at: new Date().toISOString(),
         })
         .select()
         .single();

       if (dbError) throw dbError;

       // 署名付きURL生成（1時間有効）
       const { data: signedUrl, error: urlError } = await supabase.storage
         .from('posts')
         .createSignedUploadUrl(filePath, {
           expiresIn: 3600, // 1時間
           upsert: false,
         });

       if (urlError) throw urlError;

       return new Response(
         JSON.stringify({
           media_id: fileId,
           upload_url: signedUrl.signedUrl,
           expires_at: new Date(Date.now() + 3600 * 1000).toISOString(),
         }),
         {
           headers: { 'Content-Type': 'application/json' },
         },
       );
     } catch (error) {
       console.error('Media upload URL generation error:', error);
       return new Response(
         JSON.stringify({
           error: { code: 'UPLOAD_URL_FAILED', message: 'Failed to generate upload URL' },
         }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }
   ```

2. **POST /api/v1/media/stream/direct-upload（動画用）**:

   ```typescript
   import { CloudflareStreamAPI } from '../services/cloudflareStream';

   export async function POST(request: Request) {
     const authUser = await getAuthUser(request);
     if (!authUser) {
       return new Response(
         JSON.stringify({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }),
         { status: 401, headers: { 'Content-Type': 'application/json' } },
       );
     }

     const { fileName, fileSize } = await request.json();

     // ファイルサイズ検証（動画: 100MB）
     const MAX_VIDEO_SIZE = 100 * 1024 * 1024; // 100MB
     if (fileSize > MAX_VIDEO_SIZE) {
       return new Response(
         JSON.stringify({
           error: { code: 'FILE_TOO_LARGE', message: 'Video size must be less than 100MB' },
         }),
         { status: 413, headers: { 'Content-Type': 'application/json' } },
       );
     }

     try {
       const fileId = crypto.randomUUID();

       // Cloudflare Stream直接アップロード用URL作成
       const streamResponse = await CloudflareStreamAPI.createDirectUpload({
         maxDurationSeconds: 300, // 5分
         metadata: {
           userId: authUser.id,
           mediaId: fileId,
           fileName: fileName,
         },
       });

       // media_filesレコード作成
       const { data: mediaFile, error: dbError } = await supabase
         .from('media_files')
         .insert({
           id: fileId,
           user_id: authUser.id,
           file_name: fileName,
           mime_type: 'video/mp4', // Streamが自動変換
           file_size: fileSize,
           external_id: streamResponse.uid,
           storage_provider: 'cloudflare_stream',
           status: 'uploading',
           created_at: new Date().toISOString(),
         })
         .select()
         .single();

       if (dbError) throw dbError;

       return new Response(
         JSON.stringify({
           media_id: fileId,
           upload_url: streamResponse.uploadURL,
           stream_uid: streamResponse.uid,
           expires_at: new Date(Date.now() + 3600 * 1000).toISOString(),
         }),
         {
           headers: { 'Content-Type': 'application/json' },
         },
       );
     } catch (error) {
       console.error('Stream upload error:', error);
       return new Response(
         JSON.stringify({
           error: { code: 'STREAM_UPLOAD_FAILED', message: 'Failed to initialize video upload' },
         }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }
   ```

3. **Cloudflare Stream Webhook処理**:

   ```typescript
   // pages/api/webhooks/cloudflare-stream.ts
   import { CloudflareStreamAPI } from '../../../services/cloudflareStream';

   export async function POST(request: Request) {
     // Webhook署名検証
     const signature = request.headers.get('cf-signature');
     const body = await request.text();

     if (!CloudflareStreamAPI.verifyWebhookSignature(body, signature)) {
       return new Response('Invalid signature', { status: 401 });
     }

     const event = JSON.parse(body);

     // イベントID去重処理
     const eventId = event.id;
     const { data: existingEvent } = await supabase
       .from('webhook_events')
       .select('id')
       .eq('external_id', eventId)
       .single();

     if (existingEvent) {
       return new Response('Event already processed', { status: 200 });
     }

     // イベント記録
     await supabase.from('webhook_events').insert({
       external_id: eventId,
       provider: 'cloudflare_stream',
       event_type: event.eventType,
       processed_at: new Date().toISOString(),
     });

     try {
       await processStreamEvent(event);
       return new Response('OK', { status: 200 });
     } catch (error) {
       console.error('Stream webhook processing error:', error);
       // DLQ送信処理
       await sendToDLQ(event, error);
       return new Response('Processing failed', { status: 500 });
     }
   }

   async function processStreamEvent(event: any) {
     const { uid, eventType, metadata } = event;

     switch (eventType) {
       case 'video.upload.complete':
         await supabase
           .from('media_files')
           .update({
             status: 'ready',
             processed_at: new Date().toISOString(),
             metadata: { duration: event.duration, dimensions: event.dimensions },
           })
           .eq('external_id', uid);
         break;

       case 'video.upload.failed':
         await supabase
           .from('media_files')
           .update({
             status: 'failed',
             error_message: event.errorMessage,
             processed_at: new Date().toISOString(),
           })
           .eq('external_id', uid);
         break;
     }
   }
   ```

4. **画像処理パイプライン**:

   ```typescript
   // services/imageProcessor.ts
   import sharp from 'sharp';

   export class ImageProcessor {
     static async processUploadedImage(filePath: string) {
       try {
         // Storage上の画像を取得
         const { data, error } = await supabase.storage.from('posts').download(filePath);

         if (error) throw error;

         const buffer = Buffer.from(await data.arrayBuffer());

         // EXIF除去＋最適化処理
         const processedBuffer = await sharp(buffer)
           .rotate() // EXIF orientationに基づく回転
           .removeExif() // EXIF情報除去
           .jpeg({ quality: 85, progressive: true })
           .toBuffer();

         // 処理後画像を再アップロード
         await supabase.storage.from('posts').update(filePath, processedBuffer, {
           contentType: 'image/jpeg',
           cacheControl: '31536000', // 1年
         });

         // サムネイル生成（複数サイズ）
         const thumbnailSizes = [150, 300, 600];
         for (const size of thumbnailSizes) {
           const thumbnailBuffer = await sharp(processedBuffer)
             .resize(size, size, { fit: 'cover', position: 'center' })
             .jpeg({ quality: 80 })
             .toBuffer();

           const thumbnailPath = filePath.replace(/\.[^.]+$/, `_thumb_${size}.jpg`);
           await supabase.storage.from('posts').upload(thumbnailPath, thumbnailBuffer, {
             contentType: 'image/jpeg',
             cacheControl: '31536000',
           });
         }

         return { success: true };
       } catch (error) {
         console.error('Image processing error:', error);
         throw error;
       }
     }
   }
   ```

```typescript
// services/dlq.ts
export const sendToDLQ = async (event: any, error: Error) => {
  await supabase.from('failed_webhook_events').insert({
    external_id: event.id,
    provider: 'cloudflare_stream',
    event_data: event,
    error_message: error.message,
    retry_count: 0,
    created_at: new Date().toISOString(),
  });
};
```
