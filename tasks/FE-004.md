   import React, { useState, useEffect } from 'react';

   import { useForm, Controller } from 'react-hook-form';
   import * as ImagePicker from 'expo-image-picker';
   import * as ImageManipulator from 'expo-image-manipulator';
   import { useAuthStore } from '../store/authStore';
   import { profileApi } from '../services/profileApi';
   import { TextInput, Button, LoadingSpinner } from '../components/ui';
   import { useDebounce } from '../hooks/useDebounce';

   interface ProfileFormData {
     username: string;
     display_name: string;
     bio: string;
     location: string;
     website_url: string;
     avatar_url?: string;
     header_url?: string;
   }

   interface ProfileEditScreenProps {
     navigation: any;
     route: { params: { userId: string } };
   }

   export const ProfileEditScreen: React.FC<ProfileEditScreenProps> = ({ navigation, route }) => {
     const { user } = useAuthStore();
     const [isLoading, setIsLoading] = useState(false);
     const [isUploadingAvatar, setIsUploadingAvatar] = useState(false);
     const [isUploadingHeader, setIsUploadingHeader] = useState(false);
     const [usernameAvailable, setUsernameAvailable] = useState<boolean | null>(null);

     const {
       control,
       handleSubmit,
       watch,
       setValue,
       formState: { errors, isDirty }
     } = useForm<ProfileFormData>();

     const watchedUsername = watch('username');
     const debouncedUsername = useDebounce(watchedUsername, 500);

     // プロファイル取得
     useEffect(() => {
       if (user?.id) {
         loadProfile();
       }
     }, [user?.id]);

     // Username一意性チェック
     useEffect(() => {
       if (debouncedUsername && debouncedUsername.length >= 3) {
         checkUsernameAvailability(debouncedUsername);
       }
     }, [debouncedUsername]);

     const loadProfile = async () => {
       try {
         const profile = await profileApi.getProfile(user!.id);
         setValue('username', profile.username);
         setValue('display_name', profile.display_name || '');
         setValue('bio', profile.bio || '');
         setValue('location', profile.location || '');
         setValue('website_url', profile.website_url || '');
         setValue('avatar_url', profile.avatar_url || '');
         setValue('header_url', profile.header_url || '');
       } catch (error) {
         Alert.alert('エラー', 'プロファイルの読み込みに失敗しました');
       }
     };

     const checkUsernameAvailability = async (username: string) => {
       try {
         const isAvailable = await profileApi.checkUsernameAvailability(username);
         setUsernameAvailable(isAvailable);
       } catch (error) {
         setUsernameAvailable(null);
       }
     };

     const pickImage = async (type: 'avatar' | 'header') => {
       const { status } = await ImagePicker.requestMediaLibraryPermissionsAsync();
       if (status !== 'granted') {
         Alert.alert('権限エラー', '写真ライブラリへのアクセス権限が必要です');
         return;
       }

       const result = await ImagePicker.launchImageLibraryAsync({
         mediaTypes: ImagePicker.MediaTypeOptions.Images,
         allowsEditing: true,
         aspect: type === 'avatar' ? [1, 1] : [16, 9],
         quality: 0.8,
       });

       if (!result.canceled && result.assets[0]) {
         const asset = result.assets[0];
         await uploadImage(asset.uri, type);
       }
     };

     const uploadImage = async (uri: string, type: 'avatar' | 'header') => {
       const setLoading = type === 'avatar' ? setIsUploadingAvatar : setIsUploadingHeader;
       setLoading(true);

       try {
         // 画像圧縮
         const manipResult = await ImageManipulator.manipulateAsync(
           uri,
           [
             {
               resize: {
                 width: type === 'avatar' ? 300 : 800,
                 height: type === 'avatar' ? 300 : 450,
               }
             }
           ],
           {
             compress: 0.8,
             format: ImageManipulator.SaveFormat.JPEG,
           }
         );

         // Storage アップロード
         const uploadResult = await profileApi.uploadProfileImage(manipResult.uri, type);

         // フォームにURL設定
         setValue(type === 'avatar' ? 'avatar_url' : 'header_url', uploadResult.url, { shouldDirty: true });

         Alert.alert('成功', '画像をアップロードしました');
       } catch (error) {
         console.error('Image upload error:', error);
         Alert.alert('エラー', '画像のアップロードに失敗しました');
       } finally {
         setLoading(false);
       }
     };

     const onSubmit = async (data: ProfileFormData) => {
       if (!user?.id) return;

       setIsLoading(true);
       try {
         await profileApi.updateProfile(user.id, data);
         Alert.alert('成功', 'プロファイルを更新しました', [
           { text: 'OK', onPress: () => navigation.goBack() }
         ]);
       } catch (error: any) {
         const errorMessage = error.response?.data?.error?.message || '更新に失敗しました';
         Alert.alert('エラー', errorMessage);
       } finally {
         setIsLoading(false);
       }
     };

     const validateUsername = (value: string) => {
       if (value.length < 3) return '3文字以上で入力してください';
       if (value.length > 30) return '30文字以内で入力してください';
       if (!/^[a-zA-Z0-9_-]+$/.test(value)) return '英数字、アンダースコア、ハイフンのみ使用できます';
       if (usernameAvailable === false) return 'このユーザー名は既に使用されています';
       return true;
     };

     return (
       <ScrollView className="flex-1 bg-white">
         <View className="p-4">
           {/* ヘッダー画像 */}
           <TouchableOpacity
             className="h-48 bg-gray-200 rounded-lg mb-4 items-center justify-center"
             onPress={() => pickImage('header')}
           >
             {watch('header_url') ? (
               <Image source={{ uri: watch('header_url') }} className="w-full h-full rounded-lg" />
             ) : (
               <Text className="text-gray-500">ヘッダー画像を選択</Text>
             )}
             {isUploadingHeader && (
               <View className="absolute inset-0 bg-black/50 items-center justify-center rounded-lg">
                 <LoadingSpinner color="white" />
               </View>
             )}
           </TouchableOpacity>

           {/* アバター画像 */}
           <View className="items-center mb-6">
             <TouchableOpacity
               className="w-24 h-24 bg-gray-200 rounded-full items-center justify-center"
               onPress={() => pickImage('avatar')}
             >
               {watch('avatar_url') ? (
                 <Image source={{ uri: watch('avatar_url') }} className="w-24 h-24 rounded-full" />
               ) : (
                 <Text className="text-gray-500 text-xs">アバター</Text>
               )}
               {isUploadingAvatar && (
                 <View className="absolute inset-0 bg-black/50 items-center justify-center rounded-full">
                   <LoadingSpinner color="white" size="small" />
                 </View>
               )}
             </TouchableOpacity>
           </View>

           {/* フォーム */}
           <View className="space-y-4">
             <Controller
               control={control}
               name="username"
               rules={{ validate: validateUsername }}
               render={({ field: { onChange, value } }) => (
                 <View>
                   <TextInput
                     label="ユーザー名"
                     value={value}
                     onChangeText={onChange}
                     placeholder="username"
                     autoCapitalize="none"
                     error={errors.username?.message}
                   />
                   {usernameAvailable === true && (
                     <Text className="text-green-600 text-sm mt-1">✓ 使用可能です</Text>
                   )}
                   {usernameAvailable === false && (
                     <Text className="text-red-600 text-sm mt-1">✗ 既に使用されています</Text>
                   )}
                 </View>
               )}
             />

             <Controller
               control={control}
               name="display_name"
               rules={{
                 maxLength: { value: 50, message: '50文字以内で入力してください' }
               }}
               render={({ field: { onChange, value } }) => (
                 <TextInput
                   label="表示名"
                   value={value}
                   onChangeText={onChange}
                   placeholder="表示名"
                   error={errors.display_name?.message}
                 />
               )}
             />

             <Controller
               control={control}
               name="bio"
               rules={{
                 maxLength: { value: 300, message: '300文字以内で入力してください' }
               }}
               render={({ field: { onChange, value } }) => (
                 <TextInput
                   label="自己紹介"
                   value={value}
                   onChangeText={onChange}
                   placeholder="自己紹介を入力..."
                   multiline
                   numberOfLines={4}
                   error={errors.bio?.message}
                 />
               )}
             />

             <Controller
               control={control}
               name="location"
               render={({ field: { onChange, value } }) => (
                 <TextInput
                   label="所在地"
                   value={value}
                   onChangeText={onChange}
                   placeholder="東京, 日本"
                 />
               )}
             />

             <Controller
               control={control}
               name="website_url"
               rules={{
                 pattern: {
                   value: /^https?:\/\/.+/,
                   message: '有効なURLを入力してください'
                 }
               }}
               render={({ field: { onChange, value } }) => (
                 <TextInput
                   label="ウェブサイト"
                   value={value}
                   onChangeText={onChange}
                   placeholder="https://example.com"
                   keyboardType="url"
                   autoCapitalize="none"
                   error={errors.website_url?.message}
                 />
               )}
             />
           </View>

           {/* 保存ボタン */}
           <Button
             title="保存"
             onPress={handleSubmit(onSubmit)}
             loading={isLoading}
             disabled={!isDirty || isLoading}
             className="mt-6"
           />

       </ScrollView>
     );
   };

2. **profileApi.ts**:

   ```typescript
   // services/profileApi.ts
   import { supabase } from './supabase';
   import { ProfileFormData } from '../types/profile';

   export const profileApi = {
     async getProfile(userId: string) {
       const { data, error } = await supabase
         .from('profiles')
         .select('*')
         .eq('id', userId)
         .single();

       if (error) throw error;
       return data;
     },

     async updateProfile(userId: string, profileData: ProfileFormData) {
       const { data, error } = await supabase
         .from('profiles')
         .update(profileData)
         .eq('id', userId)
         .select()
         .single();

       if (error) throw error;
       return data;
     },

     async checkUsernameAvailability(username: string): Promise<boolean> {
       const { data, error } = await supabase
         .from('profiles')
         .select('id')
         .eq('username', username)
         .single();

       return !data; // データが存在しない = 使用可能
     },

     async uploadProfileImage(uri: string, type: 'avatar' | 'header') {
       const fileExt = uri.split('.').pop();
       const fileName = `${Date.now()}.${fileExt}`;
       const filePath = `${type}s/${fileName}`;

       const response = await fetch(uri);
       const blob = await response.blob();

       const { data, error } = await supabase.storage
         .from(type === 'avatar' ? 'avatars' : 'headers')
         .upload(filePath, blob, {
           contentType: blob.type,
           cacheControl: '3600',
         });

       if (error) throw error;

       const {
         data: { publicUrl },
       } = supabase.storage.from(type === 'avatar' ? 'avatars' : 'headers').getPublicUrl(data.path);

       return { url: publicUrl };
     },
   };
   ```

3. **useDebounce Hook**:

   ```typescript
   // hooks/useDebounce.ts
   import { useState, useEffect } from 'react';

   export function useDebounce<T>(value: T, delay: number): T {
     const [debouncedValue, setDebouncedValue] = useState<T>(value);

     useEffect(() => {
       const handler = setTimeout(() => {
         setDebouncedValue(value);
       }, delay);

       return () => {
         clearTimeout(handler);
       };
     }, [value, delay]);

     return debouncedValue;
   }
   ```

## デザイン実装指針（Dieter Ramsの原則に基づく）

### 1. 良いデザインは革新的である

- **画像選択のインタラクション**: 従来のボタン形式ではなく、画像領域自体をタップ可能にし、直感的な操作を実現
- **リアルタイムフィードバック**: username一意性チェックを即座に表示し、ユーザーの待機時間を最小化
- **プログレッシブな画像読み込み**: 低解像度から高解像度へのスムーズな遷移

### 2. 良いデザインはプロダクトを有用にする

- **フォーカス管理**: 入力フィールド間の自然な移動と適切なキーボード表示
- **バリデーション戦略**: エラー時のみ表示し、成功時は控えめなフィードバック
- **保存状態の明確化**: isDirtyフラグを活用した「変更あり」の視覚的表示

### 3. 良いデザインは美しい

- **一貫性のある色彩**: `colors.neutral`パレットをベースとした単色主体のデザイン
- **タイポグラフィの階層**: `typography.title`（プロファイル情報）> `typography.body`（入力フィールド）> `typography.footnote`（ヘルプテキスト）
- **余白の調和**: 既存の24pxグリッドシステムを維持

### 4. 良いデザインはプロダクトを理解しやすくする

- **視覚的階層**: ヘッダー画像 > アバター > フォーム要素の明確な重要度表現
- **状態の可視化**: ローディング、成功、エラーの各状態を色彩と形状で区別
- **情報密度の最適化**: 必要な情報のみを適切なタイミングで表示

### 5. 良いデザインは控えめである

- **最小限のアニメーション**: 既存の`fadeAnim`や`scaleAnim`パターンを踏襲
- **装飾的要素の削除**: 不要なアイコンや装飾線を排除
- **機能的な色彩使用**: `semantic.error`は実際のエラー時のみ、`semantic.success`は完了時のみ

### 6. 良いデザインは正直である

- **透明な進捗表示**: 画像アップロード中の実際の状態を隠さず表示
- **制限事項の明示**: 文字数制限やファイルサイズ制限を事前に表示
- **エラーメッセージの具体性**: 「入力してください」ではなく具体的な問題を示す

### 7. 良いデザインは持続性がある

- **コンポーネントの再利用**: 既存の`TextInput`、`Button`、`LoadingSpinner`パターンを活用
- **スケーラブルな構造**: 将来的な項目追加に対応可能な設計
- **メンテナンス性**: 明確な責任分離（UI/ロジック/API）

### 8. 良いデザインは細部まで一貫している

- **インタラクションの統一**: `TouchableOpacity`の`scaleAnim`効果を全要素に適用
- **フィードバックの統一**: エラー、成功、待機状態の表現方法を統一
- **マイクロインタラクション**: 入力フィールドのフォーカス状態変化を滑らかに

### 9. 良いデザインは環境に配慮する

- **パフォーマンス最適化**: デバウンス機能による不要なAPI呼び出し削減
- **メモリ効率**: 画像圧縮処理による端末負荷軽減
- **ネットワーク効率**: 必要最小限のデータ転送

### 10. 良いデザインはできる限りデザインしない

- **機能的最小主義**: 必要な機能のみを最もシンプルな形で提供
- **認知負荷の最小化**: 一度に表示する情報量を制限
- **操作の直感性**: 説明不要な自然なインターフェース

### 実装時の具体的指針

#### レイアウト構成

```
ヘッダー画像領域（192px高さ）
  └─ アップロード進捗オーバーレイ
アバター領域（中央配置、-48px上オフセット）
  └─ アップロード進捗オーバーレイ
フォーム領域（24pxパディング）
  ├─ Username（リアルタイムバリデーション）
  ├─ Display Name
  ├─ Bio（4行マルチライン）
  ├─ Location
  └─ Website URL
保存ボタン領域（固定下部）
```

#### 色彩使用規則

- **ベース**: `colors.neutral[0]` (背景) / `colors.neutral[800]` (テキスト)
- **入力フィールド**: `colors.neutral[200]` (背景) / `colors.neutral[700]` (テキスト)
- **エラー状態**: `colors.semantic.error` (ボーダー・テキスト)
- **成功状態**: `colors.semantic.success` (アクセントのみ)
- **非活性状態**: 50%透明度

#### インタラクション設計

- **画像選択**: タップエリアを視覚的境界より大きく設定
- **フォーム入力**: フィールド間の自動移動は実装しない（ユーザー主導を尊重）
- **バリデーション**: 入力完了後500msでチェック開始

## 実装戦略

### 戦略概要

RovRov SNSアプリのプロファイル編集機能を、フォーム管理、リアルタイムバリデーション、メディア処理、クロスプラットフォーム対応を組み合わせた本格的な実装として実現する構造化されたアプローチ。

### 技術アプローチ

#### コア技術スタック

- **フォーム管理**: react-hook-formによる効率的な状態管理とバリデーション
- **画像処理**: expo-image-picker + expo-image-manipulatorによる包括的メディア処理
- **状態管理**: 認証はZustand、フォームはローカル状態管理
- **API層**: Supabaseとのエラーハンドリング・リトライロジック付き統合
- **バリデーション**: クライアントサイド検証とサーバー確認の組み合わせ
- **パフォーマンス**: デバウンス処理、遅延読み込み、楽観的UI更新

#### アーキテクチャ原則

- **責任の分離**: UI、ビジネスロジック、API層の明確な分離
- **型安全性**: TypeScriptによる厳密な型定義
- **再利用性**: 既存のUIコンポーネント（TextInput、Button、LoadingSpinner）の活用
- **拡張性**: 将来的な項目追加に対応可能な柔軟な設計
