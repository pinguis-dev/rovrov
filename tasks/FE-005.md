   ```typescript
   import React, { useState, useEffect, useRef } from 'react';
   import { View, Text, Alert, BackHandler } from 'react-native';
   import { useNavigation } from '@react-navigation/native';
   import AsyncStorage from '@react-native-async-storage/async-storage';
   import { useAuthStore } from '../store/authStore';
   import { postApi } from '../services/postApi';
   import { MediaSelector } from '../components/post/MediaSelector';
   import { CaptionEditor } from '../components/post/CaptionEditor';
   import { LocationPicker } from '../components/post/LocationPicker';
   import { VisibilitySettings } from '../components/post/VisibilitySettings';
   import { StepIndicator, Button } from '../components/ui';
   import { generateIdempotencyKey } from '../utils/api';

   interface PostDraft {
     id: string;
     step: number;
     mediaAssets: MediaAsset[];
     caption: string;
     tags: string[];
     pinId?: string;
     location?: {
       latitude: number;
       longitude: number;
       address?: string;
     };
     visibility: 'public' | 'friends' | 'private';
     mapVisibility: 'none' | 'approx_100m' | 'exact';
     createdAt: string;
     updatedAt: string;
   }

   interface MediaAsset {
     id: string;
     uri: string;
     type: 'image' | 'video';
     mimeType: string;
     fileSize: number;
     dimensions?: { width: number; height: number };
     uploadStatus: 'pending' | 'uploading' | 'completed' | 'failed';
     uploadProgress: number;
     mediaId?: string; // サーバーから返されるID
   }

   export const PostCreatorWizard: React.FC = () => {
     const navigation = useNavigation();
     const { user } = useAuthStore();
     const [currentStep, setCurrentStep] = useState(1);
     const [isLoading, setIsLoading] = useState(false);
     const [draft, setDraft] = useState<PostDraft | null>(null);
     const autoSaveTimer = useRef<NodeJS.Timeout>();

     const steps = [
       { title: 'メディア選択', component: MediaSelector },
       { title: 'キャプション', component: CaptionEditor },
       { title: '位置情報', component: LocationPicker },
       { title: '公開設定', component: VisibilitySettings }
     ];

     useEffect(() => {
       initializeDraft();
       setupAutoSave();
       setupBackHandler();

       return () => {
         if (autoSaveTimer.current) {
           clearInterval(autoSaveTimer.current);
         }
       };
     }, []);

     const initializeDraft = async () => {
       try {
         // 既存の下書きがあるかチェック
         const existingDraft = await AsyncStorage.getItem('post_draft');
         if (existingDraft) {
           const parsedDraft = JSON.parse(existingDraft);

           Alert.alert(
             '下書きが見つかりました',
             '前回作成中だった投稿を復元しますか？',
             [
               { text: 'いいえ', onPress: createNewDraft },
               { text: 'はい', onPress: () => restoreDraft(parsedDraft) }
             ]
           );
         } else {
           createNewDraft();
         }
       } catch (error) {
         console.error('Draft initialization error:', error);
         createNewDraft();
       }
     };

     const createNewDraft = () => {
       const newDraft: PostDraft = {
         id: generateIdempotencyKey(),
         step: 1,
         mediaAssets: [],
         caption: '',
         tags: [],
         visibility: 'public',
         mapVisibility: 'none',
         createdAt: new Date().toISOString(),
         updatedAt: new Date().toISOString()
       };

       setDraft(newDraft);
       saveDraftToStorage(newDraft);
     };

     const restoreDraft = (existingDraft: PostDraft) => {
       setDraft(existingDraft);
       setCurrentStep(existingDraft.step);
     };

     const setupAutoSave = () => {
       autoSaveTimer.current = setInterval(() => {
         if (draft) {
           saveDraftToStorage(draft);
         }
       }, 30000); // 30秒ごと
     };

     const setupBackHandler = () => {
       const backAction = () => {
         if (currentStep > 1) {
           setCurrentStep(currentStep - 1);
           return true;
         }

         // 最初のステップでバックキー押下時の確認
         Alert.alert(
           '投稿を破棄しますか？',
           '作成中の投稿は下書きとして保存されます',
           [
             { text: 'キャンセル', style: 'cancel' },
             { text: '破棄', style: 'destructive', onPress: () => navigation.goBack() }
           ]
         );
         return true;
       };

       const backHandler = BackHandler.addEventListener('hardwareBackPress', backAction);
       return () => backHandler.remove();
     };

     const saveDraftToStorage = async (draftData: PostDraft) => {
       try {
         const updatedDraft = {
           ...draftData,
           updatedAt: new Date().toISOString()
         };
         await AsyncStorage.setItem('post_draft', JSON.stringify(updatedDraft));
       } catch (error) {
         console.error('Draft save error:', error);
       }
     };

     const updateDraft = (updates: Partial<PostDraft>) => {
       if (draft) {
         const updatedDraft = { ...draft, ...updates };
         setDraft(updatedDraft);
         saveDraftToStorage(updatedDraft);
       }
     };

     const handleStepComplete = async (stepData: any) => {
       switch (currentStep) {
         case 1: // メディア選択完了
           updateDraft({
             mediaAssets: stepData.mediaAssets,
             step: 2
           });
           setCurrentStep(2);
           break;

         case 2: // キャプション入力完了
           updateDraft({
             caption: stepData.caption,
             tags: stepData.tags,
             step: 3
           });
           setCurrentStep(3);
           break;

         case 3: // 位置情報設定完了
           updateDraft({
             pinId: stepData.pinId,
             location: stepData.location,
             step: 4
           });
           setCurrentStep(4);
           break;

         case 4: // 公開設定完了
           updateDraft({
             visibility: stepData.visibility,
             mapVisibility: stepData.mapVisibility
           });
           await createPost();
           break;
       }
     };

     const createPost = async () => {
       if (!draft) return;

       setIsLoading(true);
       try {
         // メディアアップロードが未完了の場合は完了を待つ
         const incompleteUploads = draft.mediaAssets.filter(
           asset => asset.uploadStatus !== 'completed'
         );

         if (incompleteUploads.length > 0) {
           Alert.alert('アップロード中', 'メディアファイルのアップロードが完了するまでお待ちください');
           return;
         }

         // 投稿作成API呼び出し
         const postData = {
           caption: draft.caption,
           media_ids: draft.mediaAssets.map(asset => asset.mediaId).filter(Boolean),
           pin_id: draft.pinId,
           tags: draft.tags,
           visibility: draft.visibility,
           map_visibility: draft.mapVisibility,
           status: 'published'
         };

         const createdPost = await postApi.createPost(postData, draft.id);

         // 下書き削除
         await AsyncStorage.removeItem('post_draft');

         Alert.alert(
           '投稿完了',
           '投稿を作成しました',
           [
             {
               text: 'OK',
               onPress: () => {
                 navigation.navigate('Timeline');
               }
             }
           ]
         );

       } catch (error: any) {
         console.error('Post creation error:', error);
         Alert.alert(
           'エラー',
           error.response?.data?.error?.message || '投稿の作成に失敗しました'
         );
       } finally {
         setIsLoading(false);
       }
     };

     if (!draft) {
       return <LoadingScreen />;
     }

     const CurrentStepComponent = steps[currentStep - 1].component;

     return (
       <View className="flex-1 bg-white">
         {/* ヘッダー */}
         <View className="px-4 py-3 border-b border-gray-200">
           <StepIndicator
             steps={steps.map(s => s.title)}
             currentStep={currentStep}
           />
         </View>

         {/* ステップコンテンツ */}
         <View className="flex-1">
           <CurrentStepComponent
             draft={draft}
             onComplete={handleStepComplete}
             onUpdate={updateDraft}
           />
         </View>

         {/* フッター */}
         <View className="p-4 border-t border-gray-200">
           <View className="flex-row space-x-3">
             {currentStep > 1 && (
               <Button
                 title="戻る"
                 variant="outline"
                 onPress={() => setCurrentStep(currentStep - 1)}
                 className="flex-1"
               />
             )}
             <Button
               title={currentStep === 4 ? '投稿する' : '次へ'}
               onPress={() => {
                 // 各ステップのバリデーション実行は各コンポーネントで実施
               }}
               loading={isLoading}
               className="flex-1"
             />
           </View>
         </View>
       </View>
     );
   };
   ```

2. **MediaSelector.tsx**:

   ```typescript
   import React, { useState } from 'react';
   import { View, Text, TouchableOpacity, Alert, FlatList, Image } from 'react-native';
   import * as ImagePicker from 'expo-image-picker';
   import * as ImageManipulator from 'expo-image-manipulator';
   import { Video } from 'expo-av';
   import { mediaApi } from '../../services/mediaApi';
   import { ProgressBar, Icon } from '../ui';

   interface MediaSelectorProps {
     draft: PostDraft;
     onComplete: (data: { mediaAssets: MediaAsset[] }) => void;
     onUpdate: (updates: Partial<PostDraft>) => void;
   }

   export const MediaSelector: React.FC<MediaSelectorProps> = ({
     draft,
     onComplete,
     onUpdate
   }) => {
     const [mediaAssets, setMediaAssets] = useState<MediaAsset[]>(draft.mediaAssets || []);

     const requestPermissions = async () => {
       const { status: cameraStatus } = await ImagePicker.requestCameraPermissionsAsync();
       const { status: libraryStatus } = await ImagePicker.requestMediaLibraryPermissionsAsync();

       if (cameraStatus !== 'granted' || libraryStatus !== 'granted') {
         Alert.alert('権限が必要です', 'カメラとフォトライブラリへのアクセス権限を許可してください');
         return false;
       }

       return true;
     };

     const pickMediaFromLibrary = async () => {
       const hasPermissions = await requestPermissions();
       if (!hasPermissions) return;

       const result = await ImagePicker.launchImageLibraryAsync({
         mediaTypes: ImagePicker.MediaTypeOptions.All,
         allowsMultipleSelection: true,
         selectionLimit: 10 - mediaAssets.length,
         quality: 0.8,
         videoMaxDuration: 300, // 5分
       });

       if (!result.canceled) {
         processSelectedMedia(result.assets);
       }
     };

     const takePhoto = async () => {
       const hasPermissions = await requestPermissions();
       if (!hasPermissions) return;

       const result = await ImagePicker.launchCameraAsync({
         mediaTypes: ImagePicker.MediaTypeOptions.Images,
         quality: 0.8,
         exif: true, // EXIF情報を取得（位置情報抽出用）
       });

       if (!result.canceled) {
         processSelectedMedia(result.assets);
       }
     };

     const takeVideo = async () => {
       const hasPermissions = await requestPermissions();
       if (!hasPermissions) return;

       const result = await ImagePicker.launchCameraAsync({
         mediaTypes: ImagePicker.MediaTypeOptions.Videos,
         videoMaxDuration: 300, // 5分
         videoQuality: ImagePicker.VideoQuality.High,
       });

       if (!result.canceled) {
         processSelectedMedia(result.assets);
       }
     };

     const processSelectedMedia = async (assets: ImagePicker.ImagePickerAsset[]) => {
       const newAssets: MediaAsset[] = [];

       for (const asset of assets) {
         let processedAsset = { ...asset };

         // 画像の場合は圧縮処理
         if (asset.type === 'image') {
           try {
             const manipResult = await ImageManipulator.manipulateAsync(
               asset.uri,
               [
                 {
                   resize: {
                     width: Math.min(asset.width || 1920, 1920),
                     height: Math.min(asset.height || 1920, 1920),
                   }
                 }
               ],
               {
                 compress: 0.8,
                 format: ImageManipulator.SaveFormat.JPEG,
               }
             );

             processedAsset = {
               ...asset,
               uri: manipResult.uri,
               fileSize: manipResult.fileSize || asset.fileSize || 0
             };
           } catch (error) {
             console.error('Image compression error:', error);
           }
         }

         const mediaAsset: MediaAsset = {
           id: `temp_${Date.now()}_${Math.random()}`,
           uri: processedAsset.uri,
           type: asset.type === 'video' ? 'video' : 'image',
           mimeType: asset.type === 'video' ? 'video/mp4' : 'image/jpeg',
           fileSize: processedAsset.fileSize || 0,
           dimensions: {
             width: asset.width || 0,
             height: asset.height || 0
           },
           uploadStatus: 'pending',
           uploadProgress: 0
         };

         // EXIF位置情報抽出
         if (asset.exif?.GPS) {
           mediaAsset.location = {
             latitude: asset.exif.GPS.Latitude,
             longitude: asset.exif.GPS.Longitude
           };
         }

         newAssets.push(mediaAsset);

         // バックグラウンドでアップロード開始
         uploadMedia(mediaAsset);
       }

       const updatedAssets = [...mediaAssets, ...newAssets];
       setMediaAssets(updatedAssets);
       onUpdate({ mediaAssets: updatedAssets });
     };

     const uploadMedia = async (asset: MediaAsset) => {
       try {
         // アップロード状態更新
         updateAssetStatus(asset.id, 'uploading', 0);

         let uploadResult;
         if (asset.type === 'video') {
           uploadResult = await mediaApi.uploadVideo(
             asset.uri,
             asset.mimeType,
             asset.fileSize,
             (progress) => updateAssetStatus(asset.id, 'uploading', progress)
           );
         } else {
           uploadResult = await mediaApi.uploadImage(
             asset.uri,
             asset.mimeType,
             asset.fileSize,
             (progress) => updateAssetStatus(asset.id, 'uploading', progress)
           );
         }

         // アップロード完了
         updateAssetStatus(asset.id, 'completed', 100, uploadResult.mediaId);

       } catch (error) {
         console.error('Media upload error:', error);
         updateAssetStatus(asset.id, 'failed', 0);

         Alert.alert(
           'アップロードエラー',
           'メディアファイルのアップロードに失敗しました。再試行しますか？',
           [
             { text: 'スキップ', onPress: () => removeAsset(asset.id) },
             { text: '再試行', onPress: () => uploadMedia(asset) }
           ]
         );
       }
     };

     const updateAssetStatus = (
       assetId: string,
       status: MediaAsset['uploadStatus'],
       progress: number,
       mediaId?: string
     ) => {
       setMediaAssets(prev => prev.map(asset =>
         asset.id === assetId
           ? { ...asset, uploadStatus: status, uploadProgress: progress, mediaId }
           : asset
       ));
     };

     const removeAsset = (assetId: string) => {
       const updatedAssets = mediaAssets.filter(asset => asset.id !== assetId);
       setMediaAssets(updatedAssets);
       onUpdate({ mediaAssets: updatedAssets });
     };

     const canProceed = () => {
       return mediaAssets.length > 0 &&
              mediaAssets.every(asset =>
                asset.uploadStatus === 'completed' || asset.uploadStatus === 'failed'
              );
     };

     const handleNext = () => {
       if (canProceed()) {
         onComplete({ mediaAssets });
       } else {
         Alert.alert('確認', 'アップロード中のファイルがあります。しばらくお待ちください。');
       }
     };

     const renderMediaItem = ({ item }: { item: MediaAsset }) => (
       <View className="relative mr-2 mb-2">
         {item.type === 'image' ? (
           <Image source={{ uri: item.uri }} className="w-24 h-24 rounded-lg" />
         ) : (
           <Video
             source={{ uri: item.uri }}
             style={{ width: 96, height: 96, borderRadius: 8 }}
             shouldPlay={false}
             isLooping={false}
             isMuted={true}
             resizeMode="cover"
           />
         )}

         {/* アップロード進捗 */}
         {item.uploadStatus === 'uploading' && (
           <View className="absolute inset-0 bg-black/50 rounded-lg items-center justify-center">
             <ProgressBar progress={item.uploadProgress} />
             <Text className="text-white text-xs mt-1">{Math.round(item.uploadProgress)}%</Text>
           </View>
         )}

         {/* エラー表示 */}
         {item.uploadStatus === 'failed' && (
           <View className="absolute inset-0 bg-red-500/50 rounded-lg items-center justify-center">
             <Icon name="alert-circle" size={20} color="white" />
           </View>
         )}

         {/* 削除ボタン */}
         <TouchableOpacity
           className="absolute -top-2 -right-2 bg-red-500 rounded-full w-6 h-6 items-center justify-center"
           onPress={() => removeAsset(item.id)}
         >
           <Icon name="x" size={12} color="white" />
         </TouchableOpacity>
       </View>
     );

     return (
       <View className="flex-1 p-4">
         <Text className="text-xl font-bold mb-4">メディアを選択</Text>

         {/* 選択済みメディア */}
         {mediaAssets.length > 0 && (
           <View className="mb-6">
             <Text className="text-sm font-medium mb-2">選択済み ({mediaAssets.length}/10)</Text>
             <FlatList
               data={mediaAssets}
               renderItem={renderMediaItem}
               keyExtractor={(item) => item.id}
               horizontal
               showsHorizontalScrollIndicator={false}
             />
           </View>
         )}

         {/* メディア選択ボタン */}
         <View className="space-y-3">
           <TouchableOpacity
             className="bg-blue-500 p-4 rounded-lg flex-row items-center"
             onPress={pickMediaFromLibrary}
             disabled={mediaAssets.length >= 10}
           >
             <Icon name="image" size={24} color="white" />
             <Text className="text-white font-medium ml-3">ライブラリから選択</Text>
           </TouchableOpacity>

           <TouchableOpacity
             className="bg-green-500 p-4 rounded-lg flex-row items-center"
             onPress={takePhoto}
             disabled={mediaAssets.length >= 10}
           >
             <Icon name="camera" size={24} color="white" />
             <Text className="text-white font-medium ml-3">写真を撮影</Text>
           </TouchableOpacity>

           <TouchableOpacity
             className="bg-purple-500 p-4 rounded-lg flex-row items-center"
             onPress={takeVideo}
             disabled={mediaAssets.length >= 10}
           >
             <Icon name="video" size={24} color="white" />
             <Text className="text-white font-medium ml-3">動画を撮影</Text>
           </TouchableOpacity>
         </View>

         {/* 次へボタン */}
         {mediaAssets.length > 0 && (
           <View className="mt-auto">
             <Button
               title="次へ"
               onPress={handleNext}
               disabled={!canProceed()}
               className="mt-4"
             />
           </View>
         )}
       </View>
     );
   };
   ```

   // services/postApi.ts
   import { supabase } from './supabase';
   import { apiClient } from './apiClient';

   export const postApi = {
     async createPost(postData: any, idempotencyKey: string) {
       const response = await apiClient.post('/api/v1/posts', postData, {
         headers: {
           'Idempotency-Key': idempotencyKey,
           'Content-Type': 'application/json',
         },
       });

       return response.data;
     },

     async updatePost(postId: string, updateData: any) {
       const response = await apiClient.patch(`/api/v1/posts/${postId}`, updateData);
       return response.data;
     },

     async deletePost(postId: string) {
       await apiClient.delete(`/api/v1/posts/${postId}`);
     },

     async getPost(postId: string) {
       const response = await apiClient.get(`/api/v1/posts/${postId}`);
       return response.data;
     },
   };
