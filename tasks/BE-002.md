### ENUM型定義
```sql
-- 投稿ステータス
CREATE TYPE post_status AS ENUM ('draft', 'published', 'temporary', 'archived');

-- 可視性設定
CREATE TYPE visibility_type AS ENUM ('private', 'friends', 'public');

-- 地図上の可視性
CREATE TYPE map_visibility_type AS ENUM ('none', 'approx_100m', 'exact');

-- アカウント状態
CREATE TYPE profile_status AS ENUM ('active', 'suspended', 'pending_deletion');

-- フレンド申請状態
CREATE TYPE friend_status AS ENUM ('pending', 'accepted');
```
