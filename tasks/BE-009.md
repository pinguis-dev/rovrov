### 主要タスク

- [ ] GET /api/v1/timeline/featured:
  - **アルゴリズム実装**:
    - スコア計算: `score = w1*log(favorite_count+1) + w2*repost_count - decay(published_at)`
    - 時間減衰: 7日以降はスコアを減衰
    - 上限: 100件/ユーザー
  - public投稿のみ（匿名対応）
  - 並び: スコア降順 + published_atセカンダリ
- [ ] GET /api/v1/timeline/follow:
  - フォロー中ユーザーの投稿取得
  - 認証必須
- [ ] GET /api/v1/timeline/friend:
  - 友達の全投稿取得（private含む）
  - 認証必須
- [ ] 共通処理:
  - カーソルページネーション（opaque Base64）
  - ユーザー×日付グルーピング
  - N+1問題対策（JOIN最適化）

   ```typescript
   export async function GET(request: Request) {
     const url = new URL(request.url);
     const cursor = url.searchParams.get('cursor');
     const limit = Math.min(parseInt(url.searchParams.get('limit') || '20'), 50);

     try {
       let query = supabase
         .from('posts')
         .select(
           `
           id, user_id, caption, pin_id, display_point,
           visibility, status, favorite_count, repost_count,
           published_at, created_at,
           profiles!posts_user_id_fkey (
             id, username, display_name, avatar_url, status
           ),
           pins (
             id, name, address
           ),
           media_files (
             id, file_name, mime_type, storage_path, external_id,
             storage_provider
           )
         `,
         )
         .eq('visibility', 'public')
         .eq('status', 'published')
         .is('deleted_at', null)
         .order('featured_score', { ascending: false })
         .order('published_at', { ascending: false })
         .limit(limit);

       // カーソルページネーション
       if (cursor) {
         const decodedCursor = JSON.parse(Buffer.from(cursor, 'base64').toString());
         query = query
           .lt('featured_score', decodedCursor.score)
           .lt('published_at', decodedCursor.published_at);
       }

       const { data: posts, error } = await query;

       if (error) throw error;

       // アクティブユーザーのみフィルタ
       const filteredPosts = posts.filter((post) => post.profiles.status === 'active');

       // 次のカーソル生成
       const nextCursor =
         filteredPosts.length === limit
           ? Buffer.from(
               JSON.stringify({
                 score: filteredPosts[filteredPosts.length - 1].featured_score,
                 published_at: filteredPosts[filteredPosts.length - 1].published_at,
               }),
             ).toString('base64')
           : null;

       return new Response(
         JSON.stringify({
           posts: filteredPosts.map(formatPostResponse),
           next_cursor: nextCursor,
           has_more: filteredPosts.length === limit,
         }),
         {
           headers: { 'Content-Type': 'application/json' },
         },
       );
     } catch (error) {
       console.error('Featured timeline error:', error);
       return new Response(
         JSON.stringify({
           error: { code: 'TIMELINE_FETCH_FAILED', message: 'Failed to fetch timeline' },
         }),
         {
           status: 500,
           headers: { 'Content-Type': 'application/json' },
         },
       );
     }
   }
   ```

2. **Featuredアルゴリズム詳細**:

   ```sql
   -- featured_score計算用関数
   CREATE OR REPLACE FUNCTION calculate_featured_score(
     favorite_count INTEGER,
     repost_count INTEGER,
     published_at TIMESTAMPTZ
   )
   RETURNS NUMERIC AS $$
   DECLARE
     w1 NUMERIC := 1.0; -- いいね重み
     w2 NUMERIC := 2.0; -- リポスト重み
     base_score NUMERIC;
     age_hours NUMERIC;
     decay_factor NUMERIC;
   BEGIN
     -- ベーススコア計算
     base_score := w1 * LN(favorite_count + 1) + w2 * repost_count;

     -- 時間減衰計算
     age_hours := EXTRACT(EPOCH FROM (NOW() - published_at)) / 3600;

     -- 7日（168時間）以降は減衰開始
     IF age_hours > 168 THEN
       decay_factor := EXP(-(age_hours - 168) / 168); -- 指数減衰
       base_score := base_score * decay_factor;
     END IF;

     RETURN base_score;
   END;
   $$ LANGUAGE plpgsql IMMUTABLE;

   -- featured_score更新トリガー
   CREATE OR REPLACE FUNCTION update_featured_score()
   RETURNS TRIGGER AS $$
   BEGIN
     NEW.featured_score := calculate_featured_score(
       NEW.favorite_count,
       NEW.repost_count,
       NEW.published_at
     );
     RETURN NEW;
   END;
   $$ LANGUAGE plpgsql;

   CREATE TRIGGER posts_featured_score_update
     BEFORE INSERT OR UPDATE OF favorite_count, repost_count, published_at
     ON posts
     FOR EACH ROW
     EXECUTE FUNCTION update_featured_score();
   ```

   ```typescript
         .order('published_at', { ascending: false })
         .limit(limit);

       if (cursor) {
         const decodedCursor = JSON.parse(Buffer.from(cursor, 'base64').toString());

         }),
         {
           headers: { 'Content-Type': 'application/json' },
   ```

```typescript
// utils/timeline.ts
export async function getFollowingUserIds(userId: string): Promise<string[]> {
  const { data, error } = await supabase
    .from('follows')
    .select('following_id')
    .eq('follower_id', userId);

  return error ? [] : data.map((f) => f.following_id);
}

export async function filterPostsByFriendship(posts: any[], userId: string) {
  const friendsOnlyPosts = posts.filter((p) => p.visibility === 'friends' && p.user_id !== userId);

  if (friendsOnlyPosts.length === 0) {
    return posts.filter((p) => p.profiles.status === 'active');
  }

  // 友達関係確認
  const friendUserIds = await getFriendUserIds(userId);

  return posts.filter((post) => {
    if (post.profiles.status !== 'active') return false;
    if (post.user_id === userId) return true;
    if (post.visibility === 'public') return true;
    if (post.visibility === 'friends') {
      return friendUserIds.includes(post.user_id);
    }
    return false;
  });
}

export function formatPostResponse(post: any) {
  return {
    id: post.id,
    user_id: post.user_id,
    caption: post.caption,
    pin_id: post.pin_id,
    display_point: post.display_point,
    visibility: post.visibility,
    status: post.status,
    favorite_count: post.favorite_count,
    repost_count: post.repost_count,
    published_at: post.published_at,
    created_at: post.created_at,
    author: {
      id: post.profiles.id,
      username: post.profiles.username,
      display_name: post.profiles.display_name,
      avatar_url: post.profiles.avatar_url,
    },
    pin: post.pins
      ? {
          id: post.pins.id,
          name: post.pins.name,
          address: post.pins.address,
        }
      : null,
    media_files: post.media_files.map((media) => ({
      id: media.id,
      file_name: media.file_name,
      mime_type: media.mime_type,
      url: generateMediaUrl(media),
      storage_provider: media.storage_provider,
    })),
  };
}
```

## Featuredアルゴリズム詳細仕様

- スコア式の重み: w1=1.0（いいね）、w2=2.0（リポスト）
- 除外: private/friends投稿、suspended/pending_deletionユーザー
- スパム抑制: 同一ユーザーは最大3投稿まで
- 時間減衰: 7日以降は指数減衰（半減期168時間）

## 注意事項

- featured_scoreは投稿更新時に自動計算されるため、バッチ処理での定期更新も検討
- 友達関係の確認はパフォーマンスを考慮してバッチ処理で実装
- タイムラインの表示順序は一貫性を保つため、同一タイムスタンプの場合はIDでセカンダリソート
